00078         {

00087             const unsigned width(decoder->getWidth());
00088             const unsigned height(decoder->getHeight());
00089             const unsigned offset(decoder->getOffset());
00090

00093                 decoder->nextScanline();

00096                 const ValueType* scanline1 = static_cast<const ValueType*>(decoder->currentScanlineOfBand(1));
00097

00100                 AlphaRowIterator as(alpha_iterator.rowIterator());
00101

00103                 {
00104                     image_accessor.set(*scanline0, is);
00105                     scanline0 += offset;
