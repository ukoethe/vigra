00485 /** \brief Iterate over multiple images simultaneously in scan order. 
00486 
00487     The value type of this iterator is an instance of the handle class CoupledHandle. This allows to iterate over multiple arrays simultaneously. The coordinates can be accessed as a special band (index 0) in the handle. The scan-order is defined such that dimensions are iterated from front to back (first to last).
00488     
00489     Instances of this class are usually constructed by calling createCoupledIterator() .
00490 
00491     To get the type of a CoupledScanOrderIterator for arrays of a certain dimension and element types use CoupledIteratorType::type.
00492 
00493     The iterator supports all functions listed in the STL documentation for
00494         <a href="http://www.sgi.com/tech/stl/RandomAccessIterator.html">Random Access Iterators</a>.
00495 
00496     Example of use:
00497     \code
00498     using namespace vigra;
00499     MultiArray<2, double> image1(Shape2(5, 5));
00500     MultiArray<2, double> image2(Shape2(5, 5));
00501     // fill image with data ...
00502     
00503     typedef CoupledIteratorType<2, double, double>::type Iterator; // the type of the CoupledScanOrderIterator
00504     
00505     Iterator start = createCoupledIterator(image1, image2); // create coupled iterator for simultaneous iteration over image1, image2 and their coordinates
00506     Iterator end = start.getEndIterator();
00507     
00508     for (Iterator it = start; it < end; ++it) {
00509       std::cout << "coordinates: " << it.get<0>() << std::endl;
00510       std::cout << "image1: " << it.get<1>() << std::endl;
00511       std::cout << "image2: " << it.get<2>() << std::endl;
00512     }
00513     
00514     //random access:
00515     Iterator::value_type handle = start[15];
00516     std::cout << "image1: " << get<1>(handle) << std::endl;
00517     \endcode
00518     
00519     <b>\#include</b> <vigra/multi_iterator_coupled.hxx> <br/>
00520     Namespace: vigra
00521 */
