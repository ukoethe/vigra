00138 template <class DestImageIterator, class DestAccessor>
00139 void createGaborFilter(DestImageIterator destUpperLeft,
00140                        DestImageIterator destLowerRight, DestAccessor da,
00141                        double orientation, double centerFrequency,
00142                        double angularSigma, double radialSigma)
00143 {
00144     int w = int(destLowerRight.x - destUpperLeft.x);
00145     int h = int(destLowerRight.y - destUpperLeft.y);
00146 
00147     double squaredSum = 0.0;
00148     double cosTheta= VIGRA_CSTD::cos(orientation);
00149     double sinTheta= VIGRA_CSTD::sin(orientation);
00150 
00151     double radialSigma2 = radialSigma*radialSigma;
00152     double angularSigma2 = angularSigma*angularSigma;
00153 
00154     double wscale = w % 1 ?
00155                     1.0f / (w-1) :
00156                     1.0f / w;
00157     double hscale = h % 1 ?
00158                     1.0f / (h-1) :
00159                     1.0f / h;
00160 
00161     int dcX= (w+1)/2, dcY= (h+1)/2;
00162 
00163     double u, v;
00164     for ( int y=0; y<h; y++, destUpperLeft.y++ )
00165     {
00166         typename DestImageIterator::row_iterator dix = destUpperLeft.rowIterator();
00167 
00168         v = hscale * ((h - (y - dcY))%h - dcY);
00169         for ( int x=0; x<w; x++, dix++ )
00170         {
00171             u= wscale*((x - dcX + w)%w - dcX);
00172 
00173             double uu = cosTheta*u + sinTheta*v - centerFrequency;
00174             double vv = -sinTheta*u + cosTheta*v;
00175             double gabor;
00176 
00177             gabor = VIGRA_CSTD::exp(-0.5*(uu*uu / radialSigma2 + vv*vv / angularSigma2));
00178             squaredSum += gabor * gabor;
00179             da.set( gabor, dix );
00180         }
00181     }
00182     destUpperLeft.y -= h;
00183 
00184     // clear out DC value and remove it from the squared sum
00185     double dcValue = da(destUpperLeft);
00186     squaredSum -= dcValue * dcValue;
00187     da.set( 0.0, destUpperLeft );
00188 
00189     // normalize energy to one
00190     double factor = VIGRA_CSTD::sqrt(squaredSum);
00191     for ( int y=0; y<h; y++, destUpperLeft.y++ )
00192     {
00193         typename DestImageIterator::row_iterator dix = destUpperLeft.rowIterator();
00194 
00195         for ( int x=0; x<w; x++, dix++ )
00196         {
00197             da.set( da(dix) / factor, dix );
00198         }
00199     }
00200 }

00230 /** \brief Calculate sensible radial sigma for given parameters.

00232     For a brief introduction what is meant with "sensible" sigmas, see
00233     \ref angularGaborSigma().

00244 inline double radialGaborSigma(double centerFrequency)
00245 {
00246     double sfactor = 3.0 * VIGRA_CSTD::sqrt(VIGRA_CSTD::log(4.0));
00247     return centerFrequency / sfactor;
00248 }

00256 /** \brief Calculate sensible angular sigma for given parameters.
00257 
00258     "Sensible" means: If you use a range of gabor filters for feature
00259     detection, you are interested in minimal redundancy. This is hard
00260     to define but one possible try is to arrange the filters in
00261     frequency space, so that the half-peak-magnitude ellipses touch
00262     each other.
00263 
00264     To do so, you must know the number of directions (first parameter
00265     for the angular sigma function) and the center frequency of the
00266     filter you want to calculate the sigmas for.
00267 
00268     The exact formulas are:
00269     \code
00270     sigma_radial= 1/sqrt(ln(4)) * centerFrequency/3
00271     \endcode
00272 
00273     \code
00274     sigma_angular= 1/sqrt(ln(4)) * tan(pi/(directions*2))
00275                    * sqrt(8/9) * centerFrequency
00276     \endcode
00277 
00278     <b> Declaration:</b>
00279 
00280     \code
00281     namespace vigra {
00282         double angularGaborSigma(int directionCount, double centerFrequency)
00283     }
00284     \endcode
00285  */
00286 
00287 inline double angularGaborSigma(int directionCount, double centerFrequency)
00288 {
00289     return VIGRA_CSTD::tan(M_PI/directionCount/2.0) * centerFrequency
00290         * VIGRA_CSTD::sqrt(8.0 / (9 * VIGRA_CSTD::log(4.0)));
00291 }

00299 /** \brief Family of gabor filters of different scale and direction.
00300 
00301     A GaborFilterFamily can be used to quickly create a whole family
00302     of gabor filters in frequency space. Especially useful in
00303     conjunction with \ref applyFourierFilterFamily, since it's derived
00304     from \ref ImageArray.
00305 
00306     The filter parameters are chosen to make the center frequencies
00307     decrease in octaves with increasing scale indices, and to make the
00308     half-peak-magnitude ellipses touch each other to somewhat reduce
00309     redundancy in the filter answers. This is done by using \ref
00310     angularGaborSigma() and \ref radialGaborSigma(), you'll find more
00311     information there.
00312 
00313     The template parameter ImageType should be a scalar image type suitable for filling in
00314 
00315     <b>\#include</b> <vigra/gaborfilter.hxx><br/>
00316     Namespace: vigra
00317 */
00318 template <class ImageType, 
00319           class Alloc = typename ImageType::allocator_type::template rebind<ImageType>::other >
00320 class GaborFilterFamily 
00321 : public ImageArray<ImageType, Alloc>
00322 {
00323     typedef ImageArray<ImageType, Alloc> ParentClass;
00324     int scaleCount_, directionCount_;
00325     double maxCenterFrequency_;
00326 
00327 protected:
00328     void initFilters()
00329     {
00330         for(int direction= 0; direction<directionCount_; direction++)
00331             for(int scale= 0; scale<scaleCount_; scale++)
00332             {
00333                 double angle = direction * M_PI / directionCount();
00334                 double centerFrequency =
00335                     maxCenterFrequency_ / VIGRA_CSTD::pow(2.0, (double)scale);
00336                 createGaborFilter(destImageRange(this->images_[filterIndex(direction, scale)]),
00337                                   angle, centerFrequency,
00338                                   angularGaborSigma(directionCount(), centerFrequency),
00339                                   radialGaborSigma(centerFrequency));
00340             }
00341     }
00342 
00343 public:
00344     enum { stdFilterSize= 128, stdDirectionCount= 6, stdScaleCount= 4 };
00345 
00346         /** Constructs a family of gabor filters in frequency
00347             space. The filters will be calculated on construction, so
00348             it makes sense to provide good parameters right now
00349             although they can be changed later, too. If you leave them
00350             out, the defaults are a \ref directionCount of 6, a \ref
00351             scaleCount of 4 and a \ref maxCenterFrequency of
00352             3/8(=0.375).
00353         */
00354     GaborFilterFamily(const Diff2D & size,
00355                       int directionCount = stdDirectionCount, int scaleCount = stdScaleCount,
00356                       double maxCenterFrequency = 3.0/8.0,
00357                       Alloc const & alloc = Alloc())
00358         : ParentClass(directionCount*scaleCount, size, alloc),
00359           scaleCount_(scaleCount),
00360           directionCount_(directionCount),
00361           maxCenterFrequency_(maxCenterFrequency)
00362     {
00363         initFilters();
00364     }
00365 
00366         /** Convenience variant of the above constructor taking width
00367             and height separately. Also, this one serves as default
00368             constructor constructing 128x128 pixel filters.
00369          */
00370     GaborFilterFamily(int width= stdFilterSize, int height= -1,
00371                       int directionCount = stdDirectionCount, int scaleCount = stdScaleCount,
00372                       double maxCenterFrequency = 3.0/8.0,
00373                       Alloc const & alloc = Alloc())
00374         : ParentClass(directionCount*scaleCount, 
00375                       Size2D(width, height > 0 ? height : width), alloc),
00376           scaleCount_(scaleCount),
00377           directionCount_(directionCount),
00378           maxCenterFrequency_(maxCenterFrequency)
00379     {
00380         initFilters();
00381     }
00382 
00383         /** Return the index of the filter with the given direction and
00384             scale in this ImageArray. direction must in the range
00385             0..directionCount()-1 and scale in the range
00386             0..rangeCount()-1. This is useful for example if you used

00391     int filterIndex(int direction, int scale) const

00393         return scale*directionCount()+direction;

00400             <tt>filters[filterIndex(direction, scale)]</tt>.

00402     ImageType const & getFilter(int direction, int scale) const
00403     {
00404         return this->images_[filterIndex(direction, scale)];
00405     }
00406 
00407         /** Resize all filters (causing their recalculation).
00408          */
00409     virtual void resizeImages(const Diff2D &newSize)
00410     {
00411         ParentClass::resizeImages(newSize);
00412         initFilters();
00413     }
00414 
00415         /** Query the number of filter scales available.
00416          */
00417     int scaleCount() const

00419
