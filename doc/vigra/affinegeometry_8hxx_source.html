00111 /** \brief Create homogeneous matrix representing a 2D rotation about the coordinate origin.
00112  
00113     For use with \ref affineWarpImage(). Angle must be in radians.
00114 */
00115 inline
00116 linalg::TemporaryMatrix<double> rotationMatrix2DRadians(double angle)
00117 {
00118     linalg::TemporaryMatrix<double> ret(identityMatrix<double>(3));
00119     double s = std::sin(angle);
00120     double c = std::cos(angle);
00121     ret(0,0) = c;
00122     ret(1,1) = c;
00123     ret(0,1) = -s;
00124     ret(1,0) = s;
00125     return ret;
00126 }

00138 /** \brief Create homogeneous matrix representing a 2D rotation about the given point.

00143 linalg::TemporaryMatrix<double> rotationMatrix2DRadians(double angle, TinyVector<double, 2> const & center)

00165 template <int ORDER, class T, 
00166           class DestIterator, class DestAccessor>
00167 void rotateImage(SplineImageView<ORDER, T> const & src,
00168                  DestIterator id, DestAccessor dest, 
00169                  double angleInDegree, TinyVector<double, 2> const & center)
00170 {
00171     int w = src.width();
00172     int h = src.height();
00173     
00174     double angle = angleInDegree/180.0;
00175     double c = cos_pi(angle); // avoid round-off errors for simple rotations
00176     double s = sin_pi(angle);
00177     
00178     for(int y = 0; y < h; ++y, ++id.y)
00179     {
00180         typename DestIterator::row_iterator rd = id.rowIterator();
00181         double sy =  (y - center[1])*c - center[0]*s + center[1];
00182         double sx = -(y - center[1])*s - center[0]*c + center[0];
00183         for(int x=0; x < w; ++x, ++rd, sx += c, sy += s)
00184         {
00185             if(src.isInside(sx, sy))
00186                 dest.set(src(sx, sy), rd);
00187         }
00188     }
00189 }

00250 /** \brief Warp an image according to an affine transformation.
00251 
00252     <b> Declarations:</b>
00253     
00254     pass 2D array views:
00255     \code
00256     namespace vigra {
00257         template <int ORDER, class T, 
00258                   class T2, class S2,
00259                   class C>
00260         void
00261         affineWarpImage(SplineImageView<ORDER, T> const & src,
00262                         MultiArrayView<2, T2, S2> dest, 
00263                         MultiArrayView<2, double, C> const & affineMatrix);
00264     }
00265     \endcode
00266     
00267     \deprecatedAPI{affineWarpImage}
00268     pass \ref ImageIterators and \ref DataAccessors :
00269     \code
00270     namespace vigra {
00271         template <int ORDER, class T, 
00272                 class DestIterator, class DestAccessor,
00273                 class C>
00274         void affineWarpImage(SplineImageView<ORDER, T> const & src,
00275                             DestIterator dul, DestIterator dlr, DestAccessor dest, 
00276                             MultiArrayView<2, double, C> const & affineMatrix);
00277     }
00278     \endcode
00279     use argument objects in conjunction with \ref ArgumentObjectFactories :
00280     \code
00281     namespace vigra {
00282         template <int ORDER, class T, 
00283                 class DestIterator, class DestAccessor,
00284                 class C>
00285         void affineWarpImage(SplineImageView<ORDER, T> const & src,
00286                             triple<DestIterator, DestIterator, DestAccessor> dest, 
00287                             MultiArrayView<2, double, C> const & affineMatrix);
00288     }
00289     \endcode
00290     \deprecatedEnd
00291     
00292     The algorithm applies the given \a affineMatrix to the <i>destination coordinates</i> and copies
00293     the image value from the resulting source coordinates, using the given SplineImageView \a src for interpolation. 
00294     If the resulting coordinate is outside the source image, nothing will be written at that destination point.
00295     
00296     \code
00297         for all dest pixels:
00298             currentSrcCoordinate = affineMatrix * currentDestCoordinate;
00299             if src.isInside(currentSrcCoordinate):
00300                 dest[currentDestCoordinate] = src[currentSrcCoordinate]; // copy an interpolated value
00301     \endcode
00302     
00303     The matrix represents a 2-dimensional affine transform by means of homogeneous coordinates,
00304     i.e. it must be a 3x3 matrix whose last row is (0,0,1).
00305     
00306     <b> Usage:</b>
00307     
00308     <b>\#include</b> <vigra/affinegeometry.hxx><br>
00309     Namespace: vigra
00310 
00311     \code
00312     MultiArray<2, float> src(width, height);
00313     SplineImageView<3, float> spline(src);
00314     
00315     MultiArray<2, float> dest1(src.shape());
00316     
00317     // equivalent (up to round-off errors) to 
00318     //     rotateImage(spline, dest1, 45.0);
00319     TinyVector<double, 2> center((width-1.0)/2.0, (height-1.0)/2.0);
00320     affineWarpImage(spline, dest1, rotationMatrix2DDegrees(45.0, center));
00321     
00322     MultiArray<2, float> dest2(2*width-1, 2*height-1);
00323     
00324     // equivalent (up to round-off errors) to 
00325     //     resizeImageSplineInterpolation(img, dest2);
00326     // note that scaleFactor = 0.5, because we must pass the transformation from destination to source
00327     affineWarpImage(spline, dest2, scalingMatrix2D(0.5));
00328     \endcode
00329 
00330     \deprecatedUsage{affineWarpImage}

00365 template <int ORDER, class T, 
00366           class DestIterator, class DestAccessor,
00367           class C>
00368 void affineWarpImage(SplineImageView<ORDER, T> const & src,
00369                      DestIterator dul, DestIterator dlr, DestAccessor dest, 
00370                      MultiArrayView<2, double, C> const & affineMatrix)
00371 {
00372     vigra_precondition(rowCount(affineMatrix) == 3 && columnCount(affineMatrix) == 3 && 
00373                        affineMatrix(2,0) == 0.0 && affineMatrix(2,1) == 0.0 && affineMatrix(2,2) == 1.0,
00374         "affineWarpImage(): matrix doesn't represent an affine transformation with homogeneous 2D coordinates.");
00375          
00376     
00377     double w = dlr.x - dul.x;
00378     double h = dlr.y - dul.y;
00379     
00380     for(double y = 0.0; y < h; ++y, ++dul.y)
00381     {
00382         typename DestIterator::row_iterator rd = dul.rowIterator();
00383         for(double x=0.0; x < w; ++x, ++rd)
00384         {
00385             double sx = x*affineMatrix(0,0) + y*affineMatrix(0,1) + affineMatrix(0,2);
00386             double sy = x*affineMatrix(1,0) + y*affineMatrix(1,1) + affineMatrix(1,2);
00387             if(src.isInside(sx, sy))
00388                 dest.set(src(sx, sy), rd);
00389         }
00390     }
00391 }
