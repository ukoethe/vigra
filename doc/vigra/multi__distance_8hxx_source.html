00076 template <class SrcIterator, class SrcAccessor,
00077           class DestIterator, class DestAccessor >
00078 void distParabola(SrcIterator is, SrcIterator iend, SrcAccessor sa,
00079                   DestIterator id, DestAccessor da, double sigma )
00080 {
00081     // We assume that the data in the input is distance squared and treat it as such
00082     double w = iend - is;
00083     if(w <= 0)
00084         return;
00085         
00086     double sigma2 = sigma * sigma;
00087     double sigma22 = 2.0 * sigma2;
00088     
00089     typedef typename SrcAccessor::value_type SrcType;
00090     typedef DistParabolaStackEntry<SrcType> Influence;
00091     std::vector<Influence> _stack;
00092     _stack.push_back(Influence(sa(is), 0.0, 0.0, w));
00093     
00094     ++is;
00095     double current = 1.0;
00096     while(current < w )
00097     {
00098         Influence & s = _stack.back();
00099         double diff = current - s.center;
00100         double intersection = current + (sa(is) - s.prevVal - sigma2*sq(diff)) / (sigma22 * diff);
00101         
00102         if( intersection < s.left) // previous point has no influence
00103         {
00104             _stack.pop_back();
00105             if(_stack.empty())
00106             {
00107                 _stack.push_back(Influence(sa(is), 0.0, current, w));
00108             }
00109             else
00110             {
00111                 continue; // try new top of stack without advancing current
00112             }
00113         }
00114         else if(intersection < s.right)
00115         {
00116             s.right = intersection;
00117             _stack.push_back(Influence(sa(is), intersection, current, w));
00118         }
00119         ++is;
00120         ++current;
00121     }
00122 
00123     // Now we have the stack indicating which rows are influenced by (and therefore
00124     // closest to) which row. We can go through the stack and calculate the
00125     // distance squared for each element of the column.
00126     typename std::vector<Influence>::iterator it = _stack.begin();
00127     for(current = 0.0; current < w; ++current, ++id)
00128     {
00129         while( current >= it->right) 
00130             ++it; 
00131         da.set(sigma2 * sq(current - it->center) + it->prevVal, id);
00132     }
00133 }

00150 template <class SrcIterator, class SrcShape, class SrcAccessor,
00151           class DestIterator, class DestAccessor, class Array>
00152 void internalSeparableMultiArrayDistTmp(
00153                       SrcIterator si, SrcShape const & shape, SrcAccessor src,
00154                       DestIterator di, DestAccessor dest, Array const & sigmas, bool invert)
00155 {
00156     // Sigma is the spread of the parabolas. It determines the structuring element size
00157     // for ND morphology. When calculating the distance transforms, sigma is usually set to 1,
00158     // unless one wants to account for anisotropic pixel pitch
00159     enum { N =  SrcShape::static_size};
00160 
00161     // we need the Promote type here if we want to invert the image (dilation)
00162     typedef typename NumericTraits<typename DestAccessor::value_type>::RealPromote TmpType;
00163     
00164     // temporary array to hold the current line to enable in-place operation
00165     ArrayVector<TmpType> tmp( shape[0] );
00166 
00167     typedef MultiArrayNavigator<SrcIterator, N> SNavigator;
00168     typedef MultiArrayNavigator<DestIterator, N> DNavigator;
00169     
00170     
00171     // only operate on first dimension here
00172     SNavigator snav( si, shape, 0 );
00173     DNavigator dnav( di, shape, 0 );
00174 
00175     using namespace vigra::functor;
00176 
00177     for( ; snav.hasMore(); snav++, dnav++ )
00178     {
00179             // first copy source to temp for maximum cache efficiency
00180             // Invert the values if necessary. Only needed for grayscale morphology
00181             if(invert)
00182                 transformLine( snav.begin(), snav.end(), src, tmp.begin(),
00183                                typename AccessorTraits<TmpType>::default_accessor(), 
00184                                Param(NumericTraits<TmpType>::zero())-Arg1());
00185             else
00186                 copyLine( snav.begin(), snav.end(), src, tmp.begin(),
00187                           typename AccessorTraits<TmpType>::default_accessor() );
00188 
00189             detail::distParabola( srcIterRange(tmp.begin(), tmp.end(),
00190                           typename AccessorTraits<TmpType>::default_const_accessor()),
00191                           destIter( dnav.begin(), dest ), sigmas[0] );
00192     }
00193     
00194     // operate on further dimensions
00195     for( int d = 1; d < N; ++d )
00196     {
00197         DNavigator dnav( di, shape, d );
00198 
00199         tmp.resize( shape[d] );
00200         
00201 
00202         for( ; dnav.hasMore(); dnav++ )
00203         {
00204              // first copy source to temp for maximum cache efficiency
00205              copyLine( dnav.begin(), dnav.end(), dest,
00206                        tmp.begin(), typename AccessorTraits<TmpType>::default_accessor() );
00207 
00208              detail::distParabola( srcIterRange(tmp.begin(), tmp.end(),
00209                            typename AccessorTraits<TmpType>::default_const_accessor()),
00210                            destIter( dnav.begin(), dest ), sigmas[d] );
00211         }
00212     }
00213     if(invert) transformMultiArray( di, shape, dest, di, dest, -Arg1());
00214 }

00327     This function performs a squared Euclidean squared distance transform on the given
00328     multi-dimensional array. Both source and destination
00329     arrays are represented by iterators, shape objects and accessors.
00330     The destination array is required to already have the correct size.

00333     marked as zero, and non-background pixels as non-zero. If the parameter 
00334     <i>background</i> is true, then the squared distance of all background
00335     pixels to the nearest object is calculated. Otherwise, the distance of all
00336     object pixels to the nearest background pixel is calculated.

00339     This is necessary when the data have non-uniform resolution (as is common in confocal
00340     microscopy, for example).

00367 template <class SrcIterator, class SrcShape, class SrcAccessor,
00368           class DestIterator, class DestAccessor, class Array>
00369 void separableMultiDistSquared( SrcIterator s, SrcShape const & shape, SrcAccessor src,
00370                                 DestIterator d, DestAccessor dest, bool background,
00371                                 Array const & pixelPitch)
00372 {
00373     int N = shape.size();
00374 
00375     typedef typename SrcAccessor::value_type SrcType;
00376     typedef typename DestAccessor::value_type DestType;
00377     typedef typename NumericTraits<DestType>::RealPromote Real;
00378 
00379     SrcType zero = NumericTraits<SrcType>::zero();
00380 
00381     double dmax = 0.0;
00382     bool pixelPitchIsReal = false;
00383     for( int k=0; k<N; ++k)
00384     {
00385         if(int(pixelPitch[k]) != pixelPitch[k])
00386             pixelPitchIsReal = true;
00387         dmax += sq(pixelPitch[k]*shape[k]);
00388     }
00389             
00390     using namespace vigra::functor;
00391    
00392     if(dmax > NumericTraits<DestType>::toRealPromote(NumericTraits<DestType>::max()) 
00393        || pixelPitchIsReal) // need a temporary array to avoid overflows
00394     {
00395         // Threshold the values so all objects have infinity value in the beginning
00396         Real maxDist = (Real)dmax, rzero = (Real)0.0;
00397         MultiArray<SrcShape::static_size, Real> tmpArray(shape);
00398         if(background == true)
00399             transformMultiArray( s, shape, src, 
00400                                  tmpArray.traverser_begin(), typename AccessorTraits<Real>::default_accessor(),
00401                                  ifThenElse( Arg1() == Param(zero), Param(maxDist), Param(rzero) ));
00402         else
00403             transformMultiArray( s, shape, src, 
00404                                  tmpArray.traverser_begin(), typename AccessorTraits<Real>::default_accessor(),
00405                                  ifThenElse( Arg1() != Param(zero), Param(maxDist), Param(rzero) ));
00406         
00407         detail::internalSeparableMultiArrayDistTmp( tmpArray.traverser_begin(), 
00408                 shape, typename AccessorTraits<Real>::default_accessor(),
00409                 tmpArray.traverser_begin(), 
00410                 typename AccessorTraits<Real>::default_accessor(), pixelPitch);
00411         
00412         copyMultiArray(srcMultiArrayRange(tmpArray), destIter(d, dest));
00413     }
00414     else        // work directly on the destination array    
00415     {
00416         // Threshold the values so all objects have infinity value in the beginning
00417         DestType maxDist = DestType(std::ceil(dmax)), rzero = (DestType)0;
00418         if(background == true)
00419             transformMultiArray( s, shape, src, d, dest,
00420                                  ifThenElse( Arg1() == Param(zero), Param(maxDist), Param(rzero) ));
00421         else
00422             transformMultiArray( s, shape, src, d, dest, 
00423                                  ifThenElse( Arg1() != Param(zero), Param(maxDist), Param(rzero) ));
00424      
00425         detail::internalSeparableMultiArrayDistTmp( d, shape, dest, d, dest, pixelPitch);
00426     }
00427 }
