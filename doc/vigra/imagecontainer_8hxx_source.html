00447 /** \brief Class template for logarithmically tapering image pyramids.
00448 
00449     An ImagePyramid manages an array of images of the type given as
00450     template parameter, where each level has half the width and height
00451     of its predecessor.  It actually represents a sequence of pyramid

00457     via the constructor.  By default, the allocator of the
00458     <tt>ImageType</tt> is reused.
00459 
00460     <b>\#include</b> <vigra/imagecontainer.hxx> <br/>
00461     Namespace: vigra
00462 */

00464       class Alloc = typename ImageType::allocator_type::template rebind<ImageType>::other >

00466 {

00469 protected:
00470     typedef ArrayVector<ImageType, Alloc> ImageVector;

00472 
00473 public:
00474         /** the type of the contained values/images
00475          */
00476     typedef ImageType    value_type;
00477 
00478     typedef typename ImageVector::iterator iterator;
00479     typedef typename ImageVector::const_iterator const_iterator;
00480     typedef typename ImageVector::reverse_iterator reverse_iterator;
00481     typedef typename ImageVector::const_reverse_iterator const_reverse_iterator;
00482     typedef typename ImageVector::reference reference;
00483     typedef typename ImageVector::const_reference const_reference;
00484 #if !defined(_MSC_VER) || _MSC_VER >= 1300
00485     typedef typename ImageVector::pointer pointer;

00698         /** Resize this ImageArray, throwing the last images away if
00699             you make the array smaller or appending new images of the
00700             right size at the end of the array if you make it
00701             larger. (STL-Sequence interface)
00702         */
00703     void resize(int lowestLevel, int highestLevel,
00704                 const Diff2D &imageSize, int sizeAppliesToLevel = 0)
00705     {
00706         vigra_precondition(lowestLevel <= highestLevel,
00707            "ImagePyramid::resize(): lowestLevel <= highestLevel required.");
00708         vigra_precondition(lowestLevel <= sizeAppliesToLevel && sizeAppliesToLevel <= highestLevel,
00709            "ImagePyramid::resize(): sizeAppliesToLevel must be between lowest and highest level (inclusive).");
00710 
00711         ImageVector images(highestLevel - lowestLevel + 1, ImageType());
00712 
00713         images[sizeAppliesToLevel - lowestLevel].resize(imageSize);
