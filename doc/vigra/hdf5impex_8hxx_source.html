00109     /** \brief Wrapper for hid_t objects.
00110 
00111     Newly created or opened HDF5 handles are usually stored as objects of type 'hid_t'. When the handle
00112     is no longer needed, the appropriate close function must be called. However, if a function is 
00113     aborted by an exception, this is difficult to ensure. Class HDF5Handle is a smart pointer that 
00114     solves this problem by calling the close function in the destructor (This is analogous to how 
00115     VIGRA_UNIQUE_PTR calls 'delete' on the contained pointer). A pointer to the close function must be 
00116     passed to the constructor, along with an error message that is raised when creation/opening fails. 
00117     
00118     Since HDF5Handle objects are convertible to hid_t, they can be used in the code in place 
00119     of the latter.
00120 
00121     <b>Usage:</b>
00122 
00123     \code
00124     HDF5Handle file_id(H5Fopen(filename, H5F_ACC_RDWR, H5P_DEFAULT), 
00125                        &H5Fclose, 
00126                        "Error message.");
00127                        
00128     ... // use file_id in the same way as a plain hid_t object
00129     \endcode
00130 
00131     <b>\#include</b> <vigra/hdf5impex.hxx><br>
00132     Namespace: vigra

00551 file.mkdir("group");
00552 file.write("/group/dataset", out_multi_array);
00553 
00554 file.cd("/group");

01146         /** \brief Write multi arrays.

01149             \code iChunkSize = size; //size > 0 
01150             \endcode .
01151             The chunks will be hypercubes with edge length size. When <tt>iChunkSize == 0</tt>
01152             (default), the behavior depends on the <tt>compression</tt> setting: If no
01153             compression is requested, the data is written without chunking. Otherwise,
01154             chuning is required, and the chunk size is automatically selected such that
01155             each chunk contains about 300k pixels.

01158             \code compression = parameter; // 0 < parameter <= 9 
01159             \endcode
01160             where 0 stands for no compression and 9 for maximum compression.

01163             otherwise it will be interpreted as path relative to the current group.
01164 
01165             Note that the memory order between VIGRA and HDF5 files differs: VIGRA uses 
01166             Fortran-order, while HDF5 uses C-order. This means that a VIGRA MultiArray,
01167             whose indices represent the 'x'-, 'y'-, and 'z'-axis in that order, is reversed
01168             upon writing to an HDF5 file, i.e. in the file the axis order is 'z', 'y', 'x'. 
01169         */

01171     inline void write(std::string datasetName, 
01172                       const MultiArrayView<N, T, Stride> & array, 
01173                       int iChunkSize = 0, int compression = 0)
01174     {

01176         datasetName = get_absolute_path(datasetName);
01177 
01178         typename MultiArrayShape<N>::type chunkSize;
01179         for(unsigned int i = 0; i < N; i++){

01726         if(chunks[0] > 0)

01731             return res;
01732         }
01733         else if(compression > 0)
01734         {
01735             // set default chunks to enable compression 
01736             // (arbitrarily include about 300k pixels into each chunk, but make sure
01737             //  that the chunk size doesn't exceed the shape)
01738             ArrayVector<hsize_t> res(shape.begin(), shape.end());
01739             hsize_t chunk_length = (hsize_t)std::pow(300000.0, 1.0 / shape.size());
01740             for(unsigned int k=0; k < shape.size(); ++k)
01741                 if(res[k] > chunk_length)
01742                     res[k] = chunk_length;
01743             if(numBands > 1)
01744                 res.insert(res.begin(), numBands);
01745             return res;
01746         }
01747         else
01748         {
01749             return ArrayVector<hsize_t>();

01755         /** \brief takes any path and converts it into an absolute path
01756              in the current file.
01757            
01758              Elements like "." and ".." are treated as expected.
01759              Links are not supported or resolved.
01760         */
01761     inline std::string get_absolute_path(std::string path) const {
01762         // check for empty input or "." and return the current folder
01763         if(path.length() == 0 || path == "."){
01764             return currentGroupName_();
01765         }
01766 
01767         std::string str;
01768         // convert to absolute path
01769         if(relativePath_(path)){
01770             std::string cname = currentGroupName_();
01771             if (cname == "/")
01772                 str = currentGroupName_()+path;
01773             else
01774                 str = currentGroupName_()+"/"+path;
01775         }else{
01776             str = path;
01777         }
01778 
01779         // cut out "./"
01780         std::string::size_type startpos = 0;
01781         while(str.find(std::string("./"), startpos) != std::string::npos){
01782             std::string::size_type pos = str.find(std::string("./"), startpos);
01783             startpos = pos+1;
01784             // only cut if "./" is not part of "../" (see below)
01785             if(str.substr(pos-1,3) != "../"){
01786                 // cut out part of the string
01787                 str = str.substr(0,pos) + str.substr(pos+2,str.length()-pos-2);
01788                 startpos = pos;
01789             }
01790         }
01791 
01792         // cut out pairs of "bla/../"
01793         while(str.find(std::string("..")) != std::string::npos){
01794             std::string::size_type pos = str.find(std::string(".."));
01795 
01796             // find first slash after ".."
01797             std::string::size_type end = str.find("/",pos);
01798             if(end != std::string::npos){
01799                 // also include slash
01800                 end++;
01801             }else{
01802                 // no "/" after ".." --> this is a group, add a "/"
01803                 str = str + "/";
01804                 end = str.length();
01805             }
01806 
01807             // find first slash before ".."
01808             std::string::size_type prev_slash = str.rfind("/",pos);
01809             // if the root slash is the first before ".." --> Error
01810             vigra_invariant(prev_slash != 0 && prev_slash != std::string::npos,
01811                             "Error parsing path: "+str);
01812             // find second slash before ".."
01813             std::string::size_type begin = str.rfind("/",prev_slash-1);
01814 
01815             // cut out part of the string
01816             str = str.substr(0,begin+1) + str.substr(end,str.length()-end);
01817         }
01818 
01819         return str;
01820     }

02412 void HDF5File::read_(std::string datasetName, 
02413                      MultiArrayView<N, T, Stride> array, 
02414                      const hid_t datatype, const int numBandsOfType)
02415 {
02416     //Prepare to read without using HDF5ImportInfo
02417     ArrayVector<hsize_t> dimshape = getDatasetShape(datasetName);
02418 
02419     std::string errorMessage ("HDF5File::read(): Unable to open dataset '" + datasetName + "'.");
02420     HDF5Handle datasetHandle(getDatasetHandle_(datasetName), &H5Dclose, errorMessage.c_str());
02421 
02422     // the object in the HDF5 file may have one additional dimension which we 
02423     // interprete as the pixel type's bands
02424     int offset = (numBandsOfType > 1)
02425                     ? 1
02426                     : 0;
02427 
02428     vigra_precondition((N + offset ) == MultiArrayIndex(dimshape.size()), 
02429         "HDF5File::read(): Array dimension disagrees with dataset dimension.");
02430 
02431     typename MultiArrayShape<N>::type shape;
02432     for(int k=offset; k < (int)dimshape.size(); ++k)
02433         shape[k-offset] = (MultiArrayIndex)dimshape[k];
02434 
02435     vigra_precondition(shape == array.shape(),
02436                        "HDF5File::read(): Array shape disagrees with dataset shape.");
02437     if (offset)
02438         vigra_precondition(dimshape[0] == static_cast<hsize_t>(numBandsOfType),
02439                            "HDF5File::read(): Band count doesn't match destination array compound type.");
02440 
02441     herr_t status = 0;
02442     if(array.isUnstrided())
02443     {
02444         // when the array is unstrided, we can read the data directly into the array buffer
02445         status = H5Dread(datasetHandle, datatype, H5S_ALL, H5S_ALL, H5P_DEFAULT, array.data() );
02446     }
02447     else
02448     {
02449         // otherwise, we need an intermediate buffer
02450 
02451         ArrayVector<hsize_t> null(dimshape.size(), 0),
02452                              chunks(dimshape.size(), 1),
02453                              start(dimshape.size(), 0),
02454                              count(dimshape.size(), 1);
02455 
02456         HDF5Handle properties(H5Dget_create_plist(datasetHandle),
02457                              &H5Pclose, "HDF5File::read(): failed to get property list");
02458         if(H5D_CHUNKED == H5Pget_layout(properties))
02459         {
02460             // if the file is chunked, we use a buffer that matches the chunk size.
02461             H5Pget_chunk(properties, chunks.size(), chunks.data());
02462             std::reverse(chunks.begin(), chunks.end());
02463         }
02464         else
02465         {
02466             // otherwise, we compute a suitable chunk size.
02467             chunks[0] = numBandsOfType;

02469             for(unsigned int k=0; k<N; ++k)
02470             {
02471                 chunks[k+offset] = array.shape(k);
02472                 prod *= array.shape(k);
02473                 if(prod > 300000)
02474                     break;
02475             }
02476         }
02477         
02478         count[N-1-offset] = numBandsOfType;
02479         
02480         typedef typename MultiArrayShape<N>::type Shape;
02481         Shape chunkCount, chunkMaxShape;
02482         for(unsigned int k=offset; k<chunks.size(); ++k)
02483         {
02484             chunkMaxShape[k-offset] = chunks[k];
02485             chunkCount[k-offset] = (MultiArrayIndex)std::ceil(double(dimshape[k]) / chunks[k]);
02486         }
02487         
02488         typename CoupledIteratorType<N>::type chunkIter = createCoupledIterator(chunkCount),
02489                                               chunkEnd  = chunkIter.getEndIterator();
02490         for(; chunkIter != chunkEnd; ++chunkIter)
02491         {
02492             Shape chunkStart(chunkIter.point() * chunkMaxShape),
02493                   chunkStop(min(chunkStart + chunkMaxShape, array.shape()));
02494             MultiArray<N, T> buffer(chunkStop - chunkStart);
02495             
02496             for(unsigned int k=0; k<N; ++k)
02497             {
02498                 start[N-1-k] = chunkStart[k];
02499                 count[N-1-k] = buffer.shape(k);
02500             }
02501             if(offset == 1)
02502             {
02503                 start[N] = 0;
02504                 count[N] = numBandsOfType;
02505             }
02506             HDF5Handle filespace(H5Dget_space(datasetHandle),
02507                                  &H5Sclose, "HDF5File::read(): unable to create hyperslabs.");
02508             status = H5Sselect_hyperslab(filespace, H5S_SELECT_SET, start.data(), NULL, count.data(), NULL);
02509             if(status < 0)
02510                 break;
02511                 
02512             HDF5Handle dataspace(H5Screate_simple(count.size(), count.data(), NULL),
02513                                  &H5Sclose, "HDF5File::read(): unable to create hyperslabs."); 
02514             status = H5Sselect_hyperslab(dataspace, H5S_SELECT_SET, null.data(), NULL, count.data(), NULL);
02515             if(status < 0)
02516                 break;
02517                 
02518             status = H5Dread(datasetHandle, datatype, dataspace, filespace, H5P_DEFAULT, buffer.data());
02519             if(status < 0)

02521
