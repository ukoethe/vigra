00062 /** \brief Calculate the gradient energy tensor for a scalar valued image.
00063 
00064     These function calculates the gradient energy tensor (GET operator) as described in
00065     
00066     M. Felsberg, U. K&ouml;the: 
00067     <i>"GET: The Connection Between Monogenic Scale-Space and Gaussian Derivatives"</i>, 
00068     in: R. Kimmel, N. Sochen, J. Weickert (Eds.): Scale Space and PDE Methods in Computer Vision, 
00069     Proc. of Scale-Space 2005, Lecture Notes in Computer Science 3459, pp. 192-203, Heidelberg: Springer, 2005.
00070     
00071     U. K&ouml;the, M. Felsberg: 
00072     <i>"Riesz-Transforms Versus Derivatives: On the Relationship Between the Boundary Tensor and the Energy Tensor"</i>, 
00073     in: ditto, pp. 179-191.
00074 
00075     with the given filters: The derivative filter \a derivKernel is applied to the appropriate image dimensions 
00076     in turn (see the papers above for details), and the other dimension is smoothed with \a smoothKernel. 
00077     The kernels can be as small as 3x1, e.g. [0.5, 0, -0.5] and [3.0/16.0, 10.0/16.0, 3.0/16.0] respectively.  
00078     The output image must have 3 bands which will hold the
00079     tensor components in the order t11, t12 (== t21), t22. The signs of the output are adjusted for a right-handed
00080     coordinate system. Thus, orientations derived from the tensor will be in counter-clockwise (mathematically positive)
00081     order, with the x-axis at zero degrees (this is the standard in all VIGRA functions that deal with orientation).
00082     
00083     <b> Declarations:</b>
00084 
00085     pass 2D array views:
00086     \code
00087     namespace vigra {
00088         template <class T1, class S1,
00089                   class T2, class S2>
00090         void
00091         gradientEnergyTensor(MultiArrayView<2, T1, S1> const & src,
00092                              MultiArrayView<2, T2, S2> dest,
00093                              Kernel1D<double> const & derivKernel, Kernel1D<double> const & smoothKernel);
00094     }
00095     \endcode
00096

00149 template <class SrcIterator, class SrcAccessor,
00150           class DestIterator, class DestAccessor>
00151 void gradientEnergyTensor(SrcIterator supperleft, SrcIterator slowerright, SrcAccessor src,
00152                           DestIterator dupperleft, DestAccessor dest,
00153                           Kernel1D<double> const & derivKernel, Kernel1D<double> const & smoothKernel)
00154 {
00155     vigra_precondition(dest.size(dupperleft) == 3,
00156                        "gradientEnergyTensor(): output image must have 3 bands.");
00157 
00158     int w = slowerright.x - supperleft.x;
00159     int h = slowerright.y - supperleft.y;
00160     
00161     typedef typename 
00162        NumericTraits<typename SrcAccessor::value_type>::RealPromote TmpType;
00163     typedef BasicImage<TmpType> TmpImage;    
00164     TmpImage gx(w, h), gy(w, h), 
00165              gxx(w, h), gxy(w, h), gyy(w, h), 
00166              laplace(w, h), gx3(w, h), gy3(w, h);
00167     
00168     convolveImage(srcIterRange(supperleft, slowerright, src), destImage(gx), 
00169                   derivKernel, smoothKernel);
00170     convolveImage(srcIterRange(supperleft, slowerright, src), destImage(gy), 
00171                   smoothKernel, derivKernel);
00172     convolveImage(srcImageRange(gx), destImage(gxx), 
00173                   derivKernel, smoothKernel);
00174     convolveImage(srcImageRange(gx), destImage(gxy), 
00175                   smoothKernel, derivKernel);
00176     convolveImage(srcImageRange(gy), destImage(gyy), 
00177                   smoothKernel, derivKernel);
00178     combineTwoImages(srcImageRange(gxx), srcImage(gyy), destImage(laplace), 
00179                      std::plus<TmpType>());
00180     convolveImage(srcImageRange(laplace), destImage(gx3), 
00181                   derivKernel, smoothKernel);
00182     convolveImage(srcImageRange(laplace), destImage(gy3), 
00183                   smoothKernel, derivKernel);
00184     typename TmpImage::iterator gxi  = gx.begin(),
00185                                 gyi  = gy.begin(),
00186                                 gxxi = gxx.begin(),
00187                                 gxyi = gxy.begin(),
00188                                 gyyi = gyy.begin(),
00189                                 gx3i = gx3.begin(),
00190                                 gy3i = gy3.begin();
00191     for(int y = 0; y < h; ++y, ++dupperleft.y)
00192     {
00193         typename DestIterator::row_iterator d = dupperleft.rowIterator(); 
00194         for(int x = 0; x < w; ++x, ++d, ++gxi, ++gyi, ++gxxi, ++gxyi, ++gyyi, ++gx3i, ++gy3i)
00195         {
00196             dest.setComponent(sq(*gxxi) + sq(*gxyi) - *gxi * *gx3i, d, 0);
00197             dest.setComponent(- *gxyi * (*gxxi + *gyyi) + 0.5 * (*gxi * *gy3i + *gyi * *gx3i), d, 1);
00198             dest.setComponent(sq(*gxyi) + sq(*gyyi) - *gyi * *gy3i, d, 2);
00199         }
00200     }
00201 }
