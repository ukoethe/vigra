00049 template <class SrcIterator, class Shape, class SrcAccessor,
00050           class DestIterator, class DestAccessor, class Kernel>
00051 void
00052 internalResizeMultiArrayOneDimension(
00053                       SrcIterator si, Shape const & sshape, SrcAccessor src,
00054                       DestIterator di, Shape const & dshape, DestAccessor dest, 
00055                       Kernel const & spline, unsigned int d)
00056 {
00057     enum { N = 1 + SrcIterator::level };
00058 
00059     typedef typename NumericTraits<typename DestAccessor::value_type>::RealPromote TmpType;
00060 
00061     typedef MultiArrayNavigator<SrcIterator, N> SNavigator;
00062     typedef MultiArrayNavigator<DestIterator, N> DNavigator;
00063 
00064     SNavigator snav( si, sshape, d );
00065     DNavigator dnav( di, dshape, d );
00066     
00067     int ssize = sshape[d];
00068     int dsize = dshape[d];
00069 
00070     vigra_precondition(ssize > 1,
00071                  "resizeMultiArraySplineInterpolation(): "
00072                  "Source array too small.\n");
00073 
00074     Rational<int> ratio(dsize - 1, ssize - 1);
00075     Rational<int> offset(0);
00076     resampling_detail::MapTargetToSourceCoordinate mapCoordinate(ratio, offset);
00077     int period = lcm(ratio.numerator(), ratio.denominator());
00078     
00079     ArrayVector<double> const & prefilterCoeffs = spline.prefilterCoefficients();
00080     ArrayVector<Kernel1D<double> > kernels(period);
00081     createResamplingKernels(spline, mapCoordinate, kernels);
00082 
00083     // temporary array to hold the current line to enable in-place operation
00084     ArrayVector<TmpType> tmp( ssize );
00085     typename ArrayVector<TmpType>::iterator t = tmp.begin(), tend = tmp.end();
00086     typename AccessorTraits<TmpType>::default_accessor ta;
00087     
00088     for( ; snav.hasMore(); snav++, dnav++ )
00089     {
00090         // first copy source to temp for maximum cache efficiency
00091         copyLine( snav.begin(), snav.end(), src, t, ta);
00092 
00093         for(unsigned int b = 0; b < prefilterCoeffs.size(); ++b)
00094         {
00095             recursiveFilterLine(t, tend, ta, t, ta,
00096                                 prefilterCoeffs[b], BORDER_TREATMENT_REFLECT);
00097         }
00098         resamplingConvolveLine(t, tend, ta,
00099                                dnav.begin(), dnav.begin() + dsize, dest,
00100                                kernels, mapCoordinate);
00101     }
00102 }

00117 /** \brief Resize MultiArray using B-spline interpolation.
00118 
00119     <b> Declarations:</b>
00120 
00121     pass arbitrary-dimensional array views:
00122     \code
00123     namespace vigra {
00124         template <unsigned int N, class T1, class S1,
00125                                   class T2, class S2, 
00126                   class Kernel = BSpline<3, double> >

00129                                             MultiArrayView<N, T2, S2> dest,
00130                                             Kernel const & spline = BSpline<3, double>());
00131     }
00132     \endcode
00133 
00134     \deprecatedAPI{resizeMultiArraySplineInterpolation}
00135     pass \ref MultiIteratorPage "MultiIterators" and \ref DataAccessors :
00136     \code
00137     namespace vigra {
00138         template <class SrcIterator, class Shape, class SrcAccessor,
00139                   class DestIterator, class DestAccessor,
00140                   class Kernel = BSpline<3, double> >
00141         void
00142         resizeMultiArraySplineInterpolation(
00143                               SrcIterator si, Shape const & sshape, SrcAccessor src,
00144                               DestIterator di, Shape const & dshape, DestAccessor dest,
00145                               Kernel const & spline = BSpline<3, double>());
00146     }
00147     \endcode
00148     use argument objects in conjunction with \ref ArgumentObjectFactories :
00149     \code
00150     namespace vigra {
00151         template <class SrcIterator, class Shape, class SrcAccessor,
00152                   class DestIterator, class DestAccessor,
00153                   class Kernel = BSpline<3, double> >
00154         void
00155         resizeMultiArraySplineInterpolation(
00156                               triple<SrcIterator, Shape, SrcAccessor> src,
00157                               triple<DestIterator, Shape, DestAccessor> dest,
00158                               Kernel const & spline = BSpline<3, double>());
00159     }
00160     \endcode
00161     \deprecatedEnd
00162 
00163     The function implements separable spline interpolation algorithm described in
00164 
00165     M. Unser, A. Aldroubi, M. Eden, <i>"B-Spline Signal Processing"</i>
00166     IEEE Transactions on Signal Processing, vol. 41, no. 2, pp. 821-833 (part I),
00167     pp. 834-848 (part II), 1993.
00168 
00169     to obtain optimal interpolation quality and speed. You may pass the function
00170     a spline of arbitrary order (e.g. <TT>BSpline<ORDER, double></tt> or
00171     <TT>CatmullRomSpline<double></tt>). The default is a third order spline
00172     which gives a twice continuously differentiable interpolant.
00173     The implementation ensures that image values are interpolated rather
00174     than smoothed by first calling a recursive (sharpening) prefilter as
00175     described in the above paper. Then the actual interpolation is done
00176     using \ref resamplingConvolveLine().
00177 
00178     The range of both the input and output images (resp. regions)
00179     must be given. The input image must have a size of at
00180     least 4x4, the destination of at least 2x2. The scaling factors are then calculated
00181     accordingly. If the source image is larger than the destination, it
00182     is smoothed (band limited) using a recursive
00183     exponential filter. The source value_type (SrcAccessor::value_type) must
00184     be a linear algebra, i.e. it must support addition, subtraction,
00185     and multiplication (+, -, *), multiplication with a scalar
00186     real number and \ref NumericTraits "NumericTraits".
00187     The function uses accessors.
00188 
00189     <b> Usage:</b>
00190 
00191     <b>\#include</b> <vigra/multi_resize.hxx><br>
00192     Namespace: vigra
00193 
00194     \code
00195     MultiArray<3, float> src(Shape3(5, 7, 10)),
00196                          dest(Shape3(9, 13, 19)); // double the size
00197 
00198     // use default cubic spline interpolator
00199     resizeMultiArraySplineInterpolation(src, dest);
00200 
00201     // use linear interpolator
00202     resizeMultiArraySplineInterpolation(src, dest, BSpline<1, double>());
00203     \endcode
00204 
00205     \deprecatedUsage{resizeMultiArraySplineInterpolation}
00206     \code
00207     vigra::MultiArray<3, float> src(Shape3(5, 7, 10)),
00208                                 dest(Shape3(9, 13, 19)); // double the size
00209 
00210     // use linear interpolator
00211     vigra::resizeMultiArraySplineInterpolation(srcMultiArrayRange(src),
00212                destMultiArrayRange(dest), BSpline<1, double>());
00213 
00214     // use default cubic spline interpolator
00215     vigra::resizeMultiArraySplineInterpolation(
00216                srcMultiArrayRange(src),
00217                destMultiArrayRange(dest));
00218 
00219     \endcode
00220     <b> Required Interface:</b>
00221     The source and destination iterators must be compatible with \ref vigra::MultiIterator. The array value
00222     types must be models of \ref LinearSpace.

00230 void
00231 resizeMultiArraySplineInterpolation(
00232                       SrcIterator si, Shape const & sshape, SrcAccessor src,
00233                       DestIterator di, Shape const & dshape, DestAccessor dest, 
00234                       Kernel const & spline)
00235 {
00236     enum { N = 1 + SrcIterator::level };
00237     typedef typename NumericTraits<typename DestAccessor::value_type>::RealPromote TmpType;
00238     typedef MultiArray<N, TmpType> TmpArray;
00239     typedef typename AccessorTraits<TmpType>::default_accessor TmpAccessor;
00240         
00241     if(N==1)
00242     {
00243         detail::internalResizeMultiArrayOneDimension(si, sshape, src, 
00244                       di, dshape, dest, spline, 0);
00245     }
00246     else
00247     {
00248         unsigned int d = 0;
00249         Shape tmpShape(sshape);
00250         tmpShape[d] = dshape[d];
00251         MultiArray<N, TmpType> tmp(tmpShape);
00252         TmpAccessor ta;
00253         
00254         detail::internalResizeMultiArrayOneDimension(si, sshape, src, 
00255                              tmp.traverser_begin(), tmpShape, ta, spline, d);
00256         d = 1;
00257         for(; d<N-1; ++d)
00258         {
00259             tmpShape[d] = dshape[d];
00260             MultiArray<N, TmpType> dtmp(tmpShape);
00261             
00262             detail::internalResizeMultiArrayOneDimension(tmp.traverser_begin(), tmp.shape(), ta, 
00263                                   dtmp.traverser_begin(), tmpShape, ta, spline, d);
00264             dtmp.swap(tmp);
00265         }
00266         detail::internalResizeMultiArrayOneDimension(tmp.traverser_begin(), tmp.shape(), ta, 
00267                                         di, dshape, dest, spline, d);
00268     }
00269 }
