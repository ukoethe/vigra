00058 template <class KernelArray>
00059 void
00060 initGaussianPolarFilters1(double std_dev, KernelArray & k)
00061 {
00062     typedef typename KernelArray::value_type Kernel;
00063     typedef typename Kernel::iterator iterator;
00064 
00065     vigra_precondition(std_dev >= 0.0,
00066               "initGaussianPolarFilter1(): "
00067               "Standard deviation must be >= 0.");
00068 
00069     k.resize(4);
00070 
00071     int radius = (int)(4.0*std_dev + 0.5);
00072     std_dev *= 1.08179074376;
00073     double f = 1.0 / VIGRA_CSTD::sqrt(2.0 * M_PI) / std_dev;  // norm
00074     double a = 0.558868151788 / VIGRA_CSTD::pow(std_dev, 5);
00075     double b = -2.04251639729 / VIGRA_CSTD::pow(std_dev, 3);
00076     double sigma22 = -0.5 / std_dev / std_dev;
00077 
00078 
00079     for(unsigned int i=0; i<k.size(); ++i)
00080     {
00081         k[i].initExplicitly(-radius, radius);
00082         k[i].setBorderTreatment(BORDER_TREATMENT_REFLECT);
00083     }
00084 
00085     int ix;
00086     iterator c = k[0].center();
00087     for(ix=-radius; ix<=radius; ++ix)
00088     {
00089         double x = (double)ix;
00090         c[ix] = f * VIGRA_CSTD::exp(sigma22 * x * x);
00091     }
00092 
00093     c = k[1].center();
00094     for(ix=-radius; ix<=radius; ++ix)
00095     {
00096         double x = (double)ix;
00097         c[ix] = f * x * VIGRA_CSTD::exp(sigma22 * x * x);
00098     }
00099 
00100     c = k[2].center();
00101     double b2 = b / 3.0;
00102     for(ix=-radius; ix<=radius; ++ix)
00103     {
00104         double x = (double)ix;
00105         c[ix] = f * (b2 + a * x * x) * VIGRA_CSTD::exp(sigma22 * x * x);
00106     }
00107 
00108     c = k[3].center();
00109     for(ix=-radius; ix<=radius; ++ix)
00110     {
00111         double x = (double)ix;
00112         c[ix] = f * x * (b + a * x * x) * VIGRA_CSTD::exp(sigma22 * x * x);
00113     }
00114 }

00161         c[ix] = f2 * (x * x - sigma2) * VIGRA_CSTD::exp(sigma22 * x * x);

00223 template <class SrcIterator, class SrcAccessor,
00224           class DestIterator, class DestAccessor>
00225 void
00226 evenPolarFilters(SrcIterator supperleft, SrcIterator slowerright, SrcAccessor src,
00227                  DestIterator dupperleft, DestAccessor dest,
00228                  double scale, bool noLaplacian)
00229 {
00230     vigra_precondition(dest.size(dupperleft) == 3,
00231                        "evenPolarFilters(): image for even output must have 3 bands.");
00232 
00233     int w = slowerright.x - supperleft.x;
00234     int h = slowerright.y - supperleft.y;
00235 
00236     typedef typename
00237        NumericTraits<typename SrcAccessor::value_type>::RealPromote TmpType;
00238     typedef BasicImage<TinyVector<TmpType, 3> > TmpImage;
00239     typedef typename TmpImage::traverser TmpTraverser;
00240     TmpImage t(w, h);
00241 
00242     KernelArray k2;
00243     initGaussianPolarFilters2(scale, k2);
00244 
00245     // calculate filter responses for even filters
00246     VectorElementAccessor<typename TmpImage::Accessor> tmpBand(0, t.accessor());
00247     convolveImage(srcIterRange(supperleft, slowerright, src),
00248                   destImage(t, tmpBand), k2[2], k2[0]);
00249     tmpBand.setIndex(1);
00250     convolveImage(srcIterRange(supperleft, slowerright, src),
00251                   destImage(t, tmpBand), k2[1], k2[1]);
00252     tmpBand.setIndex(2);
00253     convolveImage(srcIterRange(supperleft, slowerright, src),
00254                   destImage(t, tmpBand), k2[0], k2[2]);
00255 
00256     // create even tensor from filter responses
00257     TmpTraverser tul(t.upperLeft());
00258     TmpTraverser tlr(t.lowerRight());
00259     for(; tul.y != tlr.y; ++tul.y, ++dupperleft.y)
00260     {
00261         typename TmpTraverser::row_iterator tr = tul.rowIterator();
00262         typename TmpTraverser::row_iterator trend = tr + w;
00263         typename DestIterator::row_iterator d = dupperleft.rowIterator();
00264         if(noLaplacian)
00265         {
00266             for(; tr != trend; ++tr, ++d)
00267             {
00268                 TmpType v = detail::RequiresExplicitCast<TmpType>::cast(0.5*sq((*tr)[0]-(*tr)[2]) + 2.0*sq((*tr)[1]));
00269                 dest.setComponent(v, d, 0);
00270                 dest.setComponent(0, d, 1);
00271                 dest.setComponent(v, d, 2);
00272             }
00273         }
00274         else
00275         {
00276             for(; tr != trend; ++tr, ++d)
00277             {
00278                 dest.setComponent(sq((*tr)[0]) + sq((*tr)[1]), d, 0);
00279                 dest.setComponent(-(*tr)[1] * ((*tr)[0] + (*tr)[2]), d, 1);
00280                 dest.setComponent(sq((*tr)[1]) + sq((*tr)[2]), d, 2);
00281             }
00282         }
00283     }
00284 }

00369 /** \brief Calculate Riesz transforms of the Laplacian of Gaussian.
00370 
00371     The Riesz transforms of the Laplacian of Gaussian have the following transfer
00372     functions (defined in a polar coordinate representation of the frequency domain):
00373 
00374     \f[
00375         F_{\sigma}(r, \phi)=(i \cos \phi)^n (i \sin \phi)^m r^2 e^{-r^2 \sigma^2 / 2}
00376     \f]
00377 
00378     where <i>n</i> = <tt>xorder</tt> and <i>m</i> = <tt>yorder</tt> determine th e
00379     order of the transform, and <tt>sigma > 0</tt> is the scale of the Laplacian
00380     of Gaussian. This function computes a good spatial domain approximation of
00381     these transforms for <tt>xorder + yorder <= 2</tt>. The filter responses may be used
00382     to calculate the monogenic signal or the boundary tensor.
00383 
00384     <b> Declarations:</b>
00385 
00386     pass 2D array views:
00387     \code
00388     namespace vigra {
00389         template <class T1, class S1,
00390                   class T2, class S2>
00391         void
00392         rieszTransformOfLOG(MultiArrayView<2, T1, S1> const & src,
00393                             MultiArrayView<2, T2, S2> dest,
00394                             double scale, unsigned int xorder, unsigned int yorder);
00395     }
00396     \endcode
00397 
00398     \deprecatedAPI{rieszTransformOfLOG}
00399     pass \ref ImageIterators and \ref DataAccessors :
00400     \code
00401     namespace vigra {
00402         template <class SrcIterator, class SrcAccessor,
00403                 class DestIterator, class DestAccessor>
00404         void rieszTransformOfLOG(SrcIterator supperleft, SrcIterator slowerright, SrcAccessor src,
00405                                  DestIterator dupperleft, DestAccessor dest,
00406                                  double scale, unsigned int xorder, unsigned int yorder);
00407     }
00408     \endcode
00409     use argument objects in conjunction with \ref ArgumentObjectFactories :
00410     \code
00411     namespace vigra {
00412         template <class SrcIterator, class SrcAccessor,
00413                 class DestIterator, class DestAccessor>
00414         void rieszTransformOfLOG(triple<SrcIterator, SrcIterator, SrcAccessor> src,
00415                                  pair<DestIterator, DestAccessor> dest,
00416                                  double scale, unsigned int xorder, unsigned int yorder);
00417     }
00418     \endcode
00419     \deprecatedEnd
00420 
00421     <b> Usage:</b>
00422 
00423     <b>\#include</b> <vigra/boundarytensor.hxx><br>
00424     Namespace: vigra
00425 
00426     \code
00427     MultiArrayView<2, double> impulse(17,17), res(17, 17);
00428     impulse(8,8) = 1.0;
00429 
00430     // calculate the impulse response of the first order Riesz transform in x-direction
00431     rieszTransformOfLOG(impulse, res, 2.0, 1, 0);
00432     \endcode
00433 */

00524                             destImage(t1), k3[0], k3[3]);
00525                 convolveImage(srcIterRange(supperleft, slowerright, src),
00526                             destImage(t2), k3[2], k3[1]);
00527             }
00528             combineTwoImages(srcImageRange(t1), srcImage(t2),
00529                              destIter(dupperleft, dest), std::minus<TmpType>());
00530             break;
00531         }
00532     }

00571 /** \brief Calculate the boundary tensor for a scalar valued image.
00572 
00573     These functions calculate a spatial domain approximation of
00574     the boundary tensor as described in
00575 
00576     U. K&ouml;the: <a href="http://hci.iwr.uni-heidelberg.de/people/ukoethe/papers/index.php#cite_polarfilters">
00577     <i>"Integrated Edge and Junction Detection with the Boundary Tensor"</i></a>,
00578      in: ICCV 03, Proc. of 9th Intl. Conf. on Computer Vision, Nice 2003, vol. 1,
00579      pp. 424-431, Los Alamitos: IEEE Computer Society, 2003
00580

00782 void boundaryTensor3(SrcIterator supperleft, SrcIterator slowerright, SrcAccessor sa,
00783                      DestIteratorEven dupperleft_even, DestAccessorEven even,
00784                      DestIteratorOdd dupperleft_odd, DestAccessorOdd odd,
00785                      double scale)
00786 {
00787     vigra_precondition(even.size(dupperleft_even) == 3,
00788                        "boundaryTensor3(): image for even output must have 3 bands.");
00789     vigra_precondition(odd.size(dupperleft_odd) == 3,
00790                        "boundaryTensor3(): image for odd output must have 3 bands.");
00791 
00792     detail::evenPolarFilters(supperleft, slowerright, sa,
00793                              dupperleft_even, even, scale, false);
00794 
00795     int w = slowerright.x - supperleft.x;
00796     int h = slowerright.y - supperleft.y;
00797 
00798     typedef typename
00799        NumericTraits<typename SrcAccessor::value_type>::RealPromote TmpType;
00800     typedef BasicImage<TinyVector<TmpType, 4> > TmpImage;
00801     TmpImage t1(w, h), t2(w, h);
00802 
00803     detail::KernelArray k1, k3;
00804     detail::initGaussianPolarFilters1(scale, k1);
00805     detail::initGaussianPolarFilters3(scale, k3);
00806 
00807     // calculate filter responses for odd filters
00808     VectorElementAccessor<typename TmpImage::Accessor> tmpBand(0, t1.accessor());
00809     convolveImage(srcIterRange(supperleft, slowerright, sa),
00810                   destImage(t1, tmpBand), k1[3], k1[0]);
00811     tmpBand.setIndex(1);
00812     convolveImage(srcIterRange(supperleft, slowerright, sa),
00813                   destImage(t1, tmpBand), k1[1], k1[2]);
00814     tmpBand.setIndex(2);
00815     convolveImage(srcIterRange(supperleft, slowerright, sa),
00816                   destImage(t1, tmpBand), k3[3], k3[0]);
00817     tmpBand.setIndex(3);
00818     convolveImage(srcIterRange(supperleft, slowerright, sa),
00819                   destImage(t1, tmpBand), k3[1], k3[2]);
00820 
00821     tmpBand.setIndex(0);
00822     convolveImage(srcIterRange(supperleft, slowerright, sa),
00823                   destImage(t2, tmpBand), k1[0], k1[3]);
00824     tmpBand.setIndex(1);
00825     convolveImage(srcIterRange(supperleft, slowerright, sa),
00826                   destImage(t2, tmpBand), k1[2], k1[1]);
00827     tmpBand.setIndex(2);
00828     convolveImage(srcIterRange(supperleft, slowerright, sa),
00829                   destImage(t2, tmpBand), k3[0], k3[3]);
00830     tmpBand.setIndex(3);
00831     convolveImage(srcIterRange(supperleft, slowerright, sa),
00832                   destImage(t2, tmpBand), k3[2], k3[1]);
00833 
00834     // create odd tensor from filter responses
00835     typedef typename TmpImage::traverser TmpTraverser;
00836     TmpTraverser tul1(t1.upperLeft());
00837     TmpTraverser tlr1(t1.lowerRight());
00838     TmpTraverser tul2(t2.upperLeft());
00839     for(; tul1.y != tlr1.y; ++tul1.y, ++tul2.y, ++dupperleft_odd.y)
00840     {
00841         typename TmpTraverser::row_iterator tr1 = tul1.rowIterator();
00842         typename TmpTraverser::row_iterator trend1 = tr1 + w;
00843         typename TmpTraverser::row_iterator tr2 = tul2.rowIterator();
00844         typename DestIteratorOdd::row_iterator o = dupperleft_odd.rowIterator();
00845         for(; tr1 != trend1; ++tr1, ++tr2, ++o)
00846         {
00847             TmpType d11 =  (*tr1)[0] + (*tr1)[2];
00848             TmpType d12 = -(*tr1)[1] - (*tr1)[3];
00849             TmpType d31 =  (*tr2)[0] - (*tr2)[2];
00850             TmpType d32 =  (*tr2)[1] - (*tr2)[3];
00851             TmpType d111 = 0.75 * d11 + 0.25 * d31;
00852             TmpType d112 = 0.25 * (d12 + d32);
00853             TmpType d122 = 0.25 * (d11 - d31);
00854             TmpType d222 = 0.75 * d12 - 0.25 * d32;
00855             TmpType d2 = sq(d112);
00856             TmpType d3 = sq(d122);
00857 
00858             odd.setComponent(0.25 * (sq(d111) + 2.0*d2 + d3), o, 0);
00859             odd.setComponent(0.25 * (d111*d112 + 2.0*d112*d122 + d122*d222), o, 1);
00860             odd.setComponent(0.25 * (d2 + 2.0*d3 + sq(d222)), o, 2);
00861         }
00862     }
00863 }
