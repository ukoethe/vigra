00393 /** \brief Detect and mark edges in a crack edge image using the Shen/Castan zero-crossing detector.
00394 
00395     This operator applies an exponential filter to the source image
00396     at the given <TT>scale</TT> and subtracts the result from the original image.
00397     Zero crossings are detected in the resulting difference image. Whenever the
00398     gradient at a zero crossing is greater than the given <TT>gradient_threshold</TT>,
00399     an edge point is marked (using <TT>edge_marker</TT>) in the destination image
00400     <i>between</i> the corresponding original pixels. Topologically, this means we
00401     must insert additional pixels between the original ones to represent the
00402     boundaries between the pixels (the so called zero- and one-cells, with the original
00403     pixels being two-cells). Within VIGRA, such an image is called \ref CrackEdgeImage.
00404     To allow insertion of the zero- and one-cells, the destination image must have twice the
00405     size of the original (precisely, <TT>(2*w-1)</TT> by <TT>(2*h-1)</TT> pixels). Then the algorithm
00406     proceeds as follows:
00407 
00408     \code
00409     sign of difference image     insert zero- and one-cells     resulting edge points (*)
00410     
00411                                          + . - . -                   . * . . .
00412           + - -                          . . . . .                   . * * * .
00413           + + -               =>         + . + . -           =>      . . . * .
00414           + + +                          . . . . .                   . . . * *
00415                                          + . + . +                   . . . . .
00416     \endcode
00417 
00418     Thus the edge points are marked where they actually are - in between the pixels.
00419     An important property of the resulting edge image is that it conforms to the notion
00420     of well-composedness as defined by Latecki et al., i.e. connected regions and edges
00421     obtained by a subsequent \ref Labeling do not depend on
00422     whether 4- or 8-connectivity is used.
00423     The non-edge pixels (<TT>.</TT>) in the destination image remain unchanged.
00424     The result conforms to the requirements of a \ref CrackEdgeImage. It can be further
00425     improved by the post-processing operations \ref removeShortEdges() and
00426     \ref closeGapsInCrackEdgeImage().

01404 template <class SrcIterator, class SrcAccessor, 
01405           class MagnitudeImage, class BackInsertable, class GradValue>
01406 void internalCannyFindEdgels(SrcIterator ul, SrcAccessor grad,
01407                              MagnitudeImage const & magnitude,
01408                              BackInsertable & edgels, GradValue grad_thresh)
01409 {
01410     typedef typename SrcAccessor::value_type PixelType;
01411     typedef typename PixelType::value_type ValueType;
01412 
01413     vigra_precondition(grad_thresh >= NumericTraits<GradValue>::zero(),
01414          "cannyFindEdgels(): gradient threshold must not be negative.");
01415     
01416     double t = 0.5 / VIGRA_CSTD::sin(M_PI/8.0);
01417 
01418     ul += Diff2D(1,1);
01419     for(int y=1; y<magnitude.height()-1; ++y, ++ul.y)
01420     {
01421         SrcIterator ix = ul;
01422         for(int x=1; x<magnitude.width()-1; ++x, ++ix.x)
01423         {
01424             double mag = magnitude(x, y);
01425             if(mag <= grad_thresh)
01426                    continue;
01427             ValueType gradx = grad.getComponent(ix, 0);
01428             ValueType grady = grad.getComponent(ix, 1);
01429 
01430             int dx = (int)VIGRA_CSTD::floor(gradx*t/mag + 0.5);
01431             int dy = (int)VIGRA_CSTD::floor(grady*t/mag + 0.5);
01432 
01433             int x1 = x - dx,
01434                 x2 = x + dx,
01435                 y1 = y - dy,
01436                 y2 = y + dy;
01437 
01438             double m1 = magnitude(x1, y1);
01439             double m3 = magnitude(x2, y2);
01440 
01441             if(m1 < mag && m3 <= mag)
01442             {
01443                 Edgel edgel;
01444 
01445                 // local maximum => quadratic interpolation of sub-pixel location
01446                 double del = 0.5 * (m1 - m3) / (m1 + m3 - 2.0*mag);
01447                 edgel.x = Edgel::value_type(x + dx*del);
01448                 edgel.y = Edgel::value_type(y + dy*del);
01449                 edgel.strength = Edgel::value_type(mag);
01450                 double orientation = VIGRA_CSTD::atan2(grady, gradx) + 0.5*M_PI;
01451                 if(orientation < 0.0)
01452                     orientation += 2.0*M_PI;
01453                 edgel.orientation = Edgel::value_type(orientation);
01454                 edgels.push_back(edgel);
01455             }
01456         }
01457     }
01458 }

02513           class MaskImage, class BackInsertable, class GradValue>
02514 void internalCannyFindEdgels3x3(SrcIterator ul, SrcAccessor grad,
02515                                 MaskImage const & mask,
02516                                 BackInsertable & edgels,
02517                                 GradValue grad_thresh)
02518 {
02519     typedef typename SrcAccessor::value_type PixelType;
02520     typedef typename PixelType::value_type ValueType;
02521 
02522     vigra_precondition(grad_thresh >= NumericTraits<GradValue>::zero(),
02523          "cannyFindEdgels3x3(): gradient threshold must not be negative.");
02524     
02525     ul += Diff2D(1,1);
02526     for(int y=1; y<mask.height()-1; ++y, ++ul.y)
02527     {
02528         SrcIterator ix = ul;
02529         for(int x=1; x<mask.width()-1; ++x, ++ix.x)
02530         {
02531             if(!mask(x,y))
02532                 continue;
02533 
02534             ValueType gradx = grad.getComponent(ix, 0);
02535             ValueType grady = grad.getComponent(ix, 1);
02536             double mag = hypot(gradx, grady);
02537             if(mag <= grad_thresh)
02538                    continue;
02539             double c = gradx / mag,
02540                    s = grady / mag;
02541 
02542             Matrix<double> ml(3,3), mr(3,1), l(3,1), r(3,1);
02543             l(0,0) = 1.0;
02544 
02545             for(int yy = -1; yy <= 1; ++yy)
02546             {
02547                 for(int xx = -1; xx <= 1; ++xx)
02548                 {
02549                     double u = c*xx + s*yy;
02550                     double v = norm(grad(ix, Diff2D(xx, yy)));
02551                     l(1,0) = u;
02552                     l(2,0) = u*u;
02553                     ml += outer(l);
02554                     mr += v*l;
02555                 }
02556             }
02557 
02558             linearSolve(ml, mr, r);
02559 
02560             Edgel edgel;
02561 
02562             // local maximum => quadratic interpolation of sub-pixel location
02563             double del = -r(1,0) / 2.0 / r(2,0);
02564             if(std::fabs(del) > 1.5)  // don't move by more than about a pixel diameter
02565                 del = 0.0;
02566             edgel.x = Edgel::value_type(x + c*del);
02567             edgel.y = Edgel::value_type(y + s*del);
02568             edgel.strength = Edgel::value_type(mag);
02569             double orientation = VIGRA_CSTD::atan2(grady, gradx) + 0.5*M_PI;
02570             if(orientation < 0.0)
02571                 orientation += 2.0*M_PI;
02572             edgel.orientation = Edgel::value_type(orientation);
02573             edgels.push_back(edgel);
02574         }
02575     }

02947 /** \page CrackEdgeImage Crack Edge Image
02948 
02949 Crack edges are marked <i>between</i> the pixels of an image.
02950 A Crack Edge Image is an image that represents these edges. In order
02951 to accommodate the cracks, the Crack Edge Image must be twice as large
02952 as the original image (precisely (2*w - 1) by (2*h - 1)). A Crack Edge Image
02953 can easily be derived from a binary image or from the signs of the
02954 response of a Laplacian filter. Consider the following sketch, where
02955 <TT>+</TT> encodes the foreground, <TT>-</TT> the background, and
02956 <TT>*</TT> the resulting crack edges.
02957 
02958     \code
02959 sign of difference image         insert cracks         resulting CrackEdgeImage
02960 
02961                                    + . - . -              . * . . .
02962       + - -                        . . . . .              . * * * .
02963       + + -               =>       + . + . -      =>      . . . * .
02964       + + +                        . . . . .              . . . * *
02965                                    + . + . +              . . . . .
02966     \endcode
02967 
02968 Starting from the original binary image (left), we insert crack pixels
02969 to get to the double-sized image (center). Finally, we mark all
02970 crack pixels whose non-crack neighbors have different signs as
02971 crack edge points, while all other pixels (crack and non-crack) become
02972 region pixels.
02973 
02974 <b>Requirements on a Crack Edge Image:</b>
02975 
02976 <ul>
02977     <li>Crack Edge Images have odd width and height.
02978     <li>Crack pixels have at least one odd coordinate.
02979     <li>Only crack pixels may be marked as edge points.
02980     <li>Crack pixels with two odd coordinates must be marked as edge points
02981         whenever any of their neighboring crack pixels was marked.
02982 </ul>
02983 
02984 The last two requirements ensure that both edges and regions are 4-connected.
02985 Thus, 4-connectivity and 8-connectivity yield identical connected
02986 components in a Crack Edge Image (so called <i>well-composedness</i>).
02987 This ensures that Crack Edge Images have nice topological properties
02988 (cf. L. J. Latecki: "Well-Composed Sets", Academic Press, 2000).
02989 */
