00057 /** \brief Anisotropic tensor smoothing with the hourglass filter.
00058 
00059     This function implements anisotropic tensor smoothing by an
00060     hourglass-shaped filters as described in
00061     
00062     U. K&ouml;the: <a href="http://hci.iwr.uni-heidelberg.de/people/ukoethe/papers/index.php#cite_structureTensor">
00063     <i>"Edge and Junction Detection with an Improved Structure Tensor"</i></a>, 
00064      in: Proc. of 25th DAGM Symposium, Magdeburg 2003, Lecture Notes in Computer Science 2781, 
00065      pp. 25-32, Heidelberg: Springer, 2003
00066      
00067     It is closely related to the structure tensor (see \ref structureTensor()), but
00068     replaces the linear tensor smoothing with a smoothing along edges only. 
00069     Smoothing across edges is largely suppressed. This means that the
00070     image structure is preserved much better because nearby features
00071     such as parallel edges are not blended into each other. 
00072     
00073     The hourglass filter is typically applied to a gradient tensor, i.e. the 
00074     Euclidean product of the gradient with itself, which can be obtained by a
00075     gradient operator followed with \ref vectorToTensor(), see example below. 
00076     The hourglass shape of the filter can be interpreted as indicating the likely 
00077     continuations of a local edge element. The parameter <tt>sigma</tt> determines
00078     the radius of the hourglass (i.e. how far the influence of the edge element 
00079     reaches), and <tt>rho</tt> controls its opening angle (i.e. how narrow the 
00080     edge orientation os followed). Recommended values are <tt>sigma = 1.4</tt>
00081     (or, more generally, two to three times the scale of the gradient operator
00082     used in the first step), and <tt>rho = 0.4</tt> which corresponds to an 
00083     opening angle of 22.5 degrees to either side of the edge.
00084     
00085     <b> Declarations:</b>
00086 
00087     pass 2D array views:
00088     \code
00089     namespace vigra {
00090         template <class T1, class S1,
00091                   class T2, class S2>
00092         void
00093         hourGlassFilter(MultiArrayView<2, T1, S1> const & src,
00094                         MultiArrayView<2, T2, S2> dest,
00095                         double sigma, double rho);
00096     }
00097     \endcode
00098 
00099     \deprecatedAPI{hourGlassFilter}
00100     pass \ref ImageIterators and \ref DataAccessors :
00101     \code
00102     namespace vigra {

00104                   class DestIterator, class DestAccessor>
00105         void hourGlassFilter(SrcIterator sul, SrcIterator slr, SrcAccessor src,
00106                              DestIterator dul, DestAccessor dest,
00107                              double sigma, double rho);
00108     }
00109     \endcode
00110     use argument objects in conjunction with \ref ArgumentObjectFactories :
00111     \code
00112     namespace vigra {
00113         template <class SrcIterator, class SrcAccessor,
00114                   class DestIterator, class DestAccessor>

00116         void hourGlassFilter(triple<SrcIterator, SrcIterator, SrcAccessor> s,
00117                              pair<DestIterator, DestAccessor> d,

00121     \deprecatedEnd

00154 template <class SrcIterator, class SrcAccessor,
00155           class DestIterator, class DestAccessor>
00156 void hourGlassFilter(SrcIterator sul, SrcIterator slr, SrcAccessor src,
00157                      DestIterator dul, DestAccessor dest,
00158                      double sigma, double rho)
00159 {
00160     vigra_precondition(sigma >= 0.0 && rho >= 0.0,
00161                        "hourGlassFilter(): sigma and rho must be >= 0.0");
00162     vigra_precondition(src.size(sul) == 3,
00163                        "hourGlassFilter(): input image must have 3 bands.");
00164     vigra_precondition(dest.size(dul) == 3,
00165                        "hourGlassFilter(): output image must have 3 bands.");
00166 
00167     // TODO: normalization
00168 
00169     int w = slr.x - sul.x;
00170     int h = slr.y - sul.y;
00171 
00172     double radius = VIGRA_CSTD::floor(3.0*sigma + 0.5);
00173     double sigma2 = -0.5 / sigma / sigma;
00174     double rho2 = -0.5 / rho / rho;
00175     double norm = 1.0 / (2.0 * M_PI * sigma * sigma);
00176 
00177     initImage(dul, dul+Diff2D(w,h), dest, NumericTraits<typename DestAccessor::value_type>::zero());
00178 
00179     for(int y=0; y<h; ++y, ++sul.y, ++dul.y)
00180     {
00181         SrcIterator s = sul;
00182         DestIterator d = dul;
00183         for(int x=0; x<w; ++x, ++s.x, ++d.x)
00184         {
00185             double phi = 0.5 * VIGRA_CSTD::atan2(
00186                                      2.0*src.getComponent(s,1),
00187                                      (double)src.getComponent(s,0) - src.getComponent(s,2));
00188             double u = VIGRA_CSTD::sin(phi);
00189             double v = VIGRA_CSTD::cos(phi);
00190 
00191             double x0 = x - radius < 0 ? -x : -radius;
00192             double y0 = y - radius < 0 ? -y : -radius;
00193             double x1 = x + radius >= w ? w - x - 1 : radius;
00194             double y1 = y + radius >= h ? h - y - 1 : radius;
00195 
00196             DestIterator dwul = d + Diff2D((int)x0, (int)y0);
00197 
00198             for(double yy=y0; yy <= y1; ++yy, ++dwul.y)
00199             {
00200                 typename DestIterator::row_iterator dw = dwul.rowIterator();
00201                 for(double xx=x0; xx <= x1; ++xx, ++dw)
00202                 {
00203                     double r2 = xx*xx + yy*yy;
00204                     double p  = u*xx - v*yy;
00205                     double q  = v*xx + u*yy;
00206                     double kernel = (p == 0.0) ?
00207                                       (q == 0.0) ?
00208                                        norm :
00209                                        0.0 :
00210                                        norm * VIGRA_CSTD::exp(sigma2*r2 + rho2*q*q/p/p);
00211                     dest.set(dest(dw) + kernel*src(s), dw);
00212                 }
00213             }
00214         }
00215     }
00216 }

00246 template <class SrcIterator, class SrcAccessor,
00247           class DestIterator, class DestAccessor>
00248 void ellipticGaussian(SrcIterator sul, SrcIterator slr, SrcAccessor src,
00249                       DestIterator dul, DestAccessor dest,
00250                       double sigmax, double sigmin)
00251 {
00252     vigra_precondition(sigmax >= sigmin && sigmin >= 0.0,
00253                        "ellipticGaussian(): "
00254                        "sigmax >= sigmin and sigmin >= 0.0 required");
00255 
00256     int w = slr.x - sul.x;
00257     int h = slr.y - sul.y;
00258 
00259     double radius = VIGRA_CSTD::floor(3.0*sigmax + 0.5);
00260     double sigmin2 = -0.5 / sigmin / sigmin;
00261     double norm = 1.0 / (2.0 * M_PI * sigmin * sigmax);
00262 
00263     initImage(dul, dul+Diff2D(w,h), dest, NumericTraits<typename DestAccessor::value_type>::zero());
00264 
00265     for(int y=0; y<h; ++y, ++sul.y, ++dul.y)
00266     {
00267         SrcIterator s = sul;
00268         DestIterator d = dul;
00269         for(int x=0; x<w; ++x, ++s.x, ++d.x)
00270         {
00271             typedef typename 
00272                NumericTraits<typename SrcAccessor::component_type>::RealPromote TmpType;
00273             TmpType d1 = src.getComponent(s,0) + src.getComponent(s,2);
00274             TmpType d2 = src.getComponent(s,0) - src.getComponent(s,2);
00275             TmpType d3 = 2.0 * src.getComponent(s,1);
00276             TmpType d4 = VIGRA_CSTD::sqrt(sq(d2) + sq(d3));
00277             TmpType excentricity = 1.0 - (d1 - d4) / (d1 + d4);
00278             double sigmax2 = -0.5 / sq((sigmax - sigmin)*excentricity + sigmin);
00279             
00280             double phi = 0.5 * VIGRA_CSTD::atan2(d3, d2);
00281             double u = VIGRA_CSTD::sin(phi);
00282             double v = VIGRA_CSTD::cos(phi);
00283 
00284             double x0 = x - radius < 0 ? -x : -radius;
00285             double y0 = y - radius < 0 ? -y : -radius;
00286             double x1 = x + radius >= w ? w - x - 1 : radius;
00287             double y1 = y + radius >= h ? h - y - 1 : radius;
00288 
00289             DestIterator dwul = d + Diff2D((int)x0, (int)y0);
00290 
00291             for(double yy=y0; yy <= y1; ++yy, ++dwul.y)
00292             {
00293                 typename DestIterator::row_iterator dw = dwul.rowIterator();
00294                 for(double xx=x0; xx <= x1; ++xx, ++dw)
00295                 {
00296                     double p  = u*xx - v*yy;
00297                     double q  = v*xx + u*yy;
00298                     double kernel = norm * VIGRA_CSTD::exp(sigmax2*p*p + sigmin2*q*q);
00299                     dest.set(dest(dw) + kernel*src(s), dw);
00300                 }
00301             }
00302         }
00303     }
00304 }

00334 class FoerstnerKernelBase
00335 {
00336   public:
00337     typedef double ResultType;
00338     typedef double WeightType;
00339     typedef DVector2Image::value_type VectorType;
00340     typedef VectorType::value_type    ValueType;
00341   
00342     FoerstnerKernelBase(double scale, bool ringShaped = false)
00343     : radius_((int)(3.0*scale+0.5)),
00344       weights_(2*radius_+1, 2*radius_+1),
00345       vectors_(2*radius_+1, 2*radius_+1)
00346     {
00347         double norm = 1.0 / (2.0 * M_PI * scale * scale);
00348         double s2 = -0.5 / scale / scale;
00349         
00350         for(int y = -radius_; y <= radius_; ++y)
00351         {
00352             for(int x = -radius_; x <= radius_; ++x)
00353             {
00354                 double d2 = x*x + y*y;
00355                 double d = VIGRA_CSTD::sqrt(d2);
00356                 vectors_(x+radius_,y+radius_) = d != 0.0 ?
00357                                                   VectorType(x/d, -y/d) :
00358                                                   VectorType(ValueType(0), ValueType(0));
00359                 weights_(x+radius_,y+radius_) = ringShaped ? 
00360                                        norm * d2 * VIGRA_CSTD::exp(d2 * s2) :
00361                                        norm * VIGRA_CSTD::exp(d2 * s2);
00362             }
00363         }
00364     }   
00365     
00366     ResultType operator()(int x, int y, VectorType const &) const
00367     {
00368         // isotropic filtering
00369         return weights_(radius_, radius_);
00370     }
00371 
00372     int radius_;
00373     DImage weights_;
00374     DVector2Image vectors_;
00375 };

00560 template <class SrcIterator, class SrcAccessor,
00561           class DestIterator, class DestAccessor,
00562           class Kernel>
00563 void orientedTrigonometricFilter(SrcIterator sul, SrcIterator slr, SrcAccessor src,
00564                     DestIterator dul, DestAccessor dest,
00565                     Kernel const & kernel)
00566 {
00567     vigra_precondition(src.size(sul) == 2,
00568                        "orientedTrigonometricFilter(): input image must have 2 bands.");
00569     vigra_precondition(dest.size(dul) == 3,
00570                        "orientedTrigonometricFilter(): output image must have 3 bands.");
00571 
00572     int w = slr.x - sul.x;
00573     int h = slr.y - sul.y;
00574     int radius = kernel.radius_;
00575     
00576     typedef typename SrcAccessor::value_type VectorType;
00577     typedef typename DestAccessor::value_type TensorType;
00578 
00579     initImage(dul, dul+Diff2D(w,h), dest, NumericTraits<TensorType>::zero());
00580 
00581     for(int y=0; y<h; ++y, ++sul.y, ++dul.y)
00582     {
00583         SrcIterator s = sul;
00584         DestIterator d = dul;
00585         for(int x=0; x<w; ++x, ++s.x, ++d.x)
00586         {
00587             int x0 = x - radius < 0 ? -x : -radius;
00588             int y0 = y - radius < 0 ? -y : -radius;
00589             int x1 = x + radius >= w ? w - x - 1 : radius;
00590             int y1 = y + radius >= h ? h - y - 1 : radius;
00591 
00592             VectorType v(src(s));
00593             TensorType t(sq(v[0]), v[0]*v[1], sq(v[1]));
00594             double sqMag = t[0] + t[2];
00595             double mag = VIGRA_CSTD::sqrt(sqMag);
00596             if(mag != 0.0)
00597                 v /= mag;
00598             else
