00059 template <class Graph, class T1Map, class T2Map>
00060 void
00061 prepareWatersheds(Graph const & g, 
00062                   T1Map const & data,
00063                   T2Map & lowestNeighborIndex)
00064 {
00065     typedef typename Graph::NodeIt    graph_scanner;
00066     typedef typename Graph::OutArcIt  neighbor_iterator;
00067 
00068     for (graph_scanner node(g); node != INVALID; ++node) 
00069     {
00070         typename T1Map::value_type lowestValue  = data[*node];
00071         typename T2Map::value_type lowestIndex  = -1;
00072 
00073         for(neighbor_iterator arc(g, node); arc != INVALID; ++arc) 
00074         {
00075             if(data[g.target(*arc)] <= lowestValue)
00076             {
00077                 lowestValue = data[g.target(*arc)];
00078                 lowestIndex = arc.neighborIndex();
00079             }
00080         }
00081         lowestNeighborIndex[*node] = lowestIndex;
00082     }
00083 }

00085 template <class Graph, class T1Map, class T2Map, class T3Map>
00086 typename T2Map::value_type
00087 unionFindWatersheds(Graph const & g,
00088                     T1Map const & data, 
00089                     T2Map const & lowestNeighborIndex,
00090                     T3Map & labels)
00091 {
00092     typedef typename Graph::Node          Node;
00093     typedef typename Graph::NodeIt        graph_scanner;
00094     typedef typename Graph::OutBackArcIt  neighbor_iterator;
00095     typedef typename T1Map::value_type    DataType;
00096     typedef typename T3Map::value_type    LabelType;
00097 
00098     vigra::detail::UnionFindArray<LabelType>  regions;
00099 
00100     // pass 1: find connected components
00101     for (graph_scanner node(g); node != INVALID; ++node) 
00102     {
00103         // define tentative label for current node
00104         LabelType currentLabel = regions.nextFreeLabel();
00105         bool hasPlateauNeighbor = false;
00106         
00107         for (neighbor_iterator arc(g, node); arc != INVALID; ++arc)
00108         {
00109             // merge regions if current target is center's lowest neighbor or vice versa
00110             if(lowestNeighborIndex[*node] == arc.neighborIndex() || 
00111                lowestNeighborIndex[g.target(*arc)] == g.oppositeIndex(arc.neighborIndex()))
00112             {
00113                 if(data[*node] == data[g.target(*arc)])
00114                     hasPlateauNeighbor = true;
00115                 LabelType neighborLabel = regions[labels[g.target(*arc)]];
00116                 currentLabel = regions.makeUnion(neighborLabel, currentLabel);
00117             }
00118         }
00119         
00120         if(hasPlateauNeighbor)
00121         {
00122             // we are on a plateau => link all plateau points
00123             for (neighbor_iterator arc(g, node); arc != INVALID; ++arc)
00124             {
00125                 if(data[*node] == data[g.target(*arc)])
00126                 {
00127                     LabelType neighborLabel = regions[labels[g.target(*arc)]];
00128                     currentLabel = regions.makeUnion(neighborLabel, currentLabel);
00129                 }
00130             }
00131         }
00132         
00133         // set label of current node
00134         labels[*node] = regions.finalizeLabel(currentLabel);
00135     }
00136     
00137     LabelType count = regions.makeContiguous();
00138 
00139     // pass 2: make component labels contiguous
00140     for (graph_scanner node(g); node != INVALID; ++node) 
00141     {
00142         labels[*node] = regions[labels[*node]];
00143     }
00144     return count;
00145 }

00147 template <class Graph, class T1Map, class T2Map>
00148 typename T2Map::value_type
00149 generateWatershedSeeds(Graph const & g, 
00150                        T1Map const & data,
00151                        T2Map & seeds,
00152                        SeedOptions const & options = SeedOptions())
00153 {
00154     typedef typename T1Map::value_type DataType;
00155     typedef unsigned char MarkerType;
00156     
00157     typename Graph::template NodeMap<MarkerType>  minima(g);
00158     
00159     if(options.mini == SeedOptions::LevelSets)
00160     {
00161         vigra_precondition(options.thresholdIsValid<DataType>(),
00162             "generateWatershedSeeds(): SeedOptions.levelSets() must be specified with threshold.");
00163     
00164         using namespace multi_math;
00165         minima = data <= DataType(options.thresh);
00166     }
00167     else
00168     {
00169         DataType threshold = options.thresholdIsValid<DataType>()
00170                                 ? options.thresh
00171                                 : NumericTraits<DataType>::max();
00172         
00173         if(options.mini == SeedOptions::ExtendedMinima)
00174             extendedLocalMinMaxGraph(g, data, minima, MarkerType(1), threshold, 
00175                                      std::less<DataType>(), std::equal_to<DataType>(), true);
00176         else
00177             localMinMaxGraph(g, data, minima, MarkerType(1), threshold, 
00178                              std::less<DataType>(), true);
00179     }
00180     return labelGraphWithBackground(g, minima, seeds, MarkerType(0), std::equal_to<MarkerType>());

00184 template <class Graph, class T1Map, class T2Map>
00185 typename T2Map::value_type 
00186 seededWatersheds(Graph const & g, 
00187                  T1Map const & data,
00188                  T2Map & labels,
00189                  WatershedOptions const & options)
00190 {
00191     typedef typename Graph::Node        Node;
00192     typedef typename Graph::NodeIt      graph_scanner;
00193     typedef typename Graph::OutArcIt    neighbor_iterator;
00194     typedef typename T1Map::value_type  CostType;
00195     typedef typename T2Map::value_type  LabelType;
00196 
00197     PriorityQueue<Node, CostType, true> pqueue;
00198     
00199     bool keepContours = ((options.terminate & KeepContours) != 0);
00200     LabelType maxRegionLabel = 0;
00201     
00202     for (graph_scanner node(g); node != INVALID; ++node) 
00203     {
00204         LabelType label = labels[*node];
00205         if(label != 0)
00206         {
00207             if(maxRegionLabel < label)
00208                 maxRegionLabel = label;
00209                 
00210             for (neighbor_iterator arc(g, node); arc != INVALID; ++arc)
00211             {
00212                 if(labels[g.target(*arc)] == 0)
00213                 {
00214                     // register all seeds that have an unlabeled neighbor
00215                     if(label == options.biased_label)
00216                         pqueue.push(*node, data[*node] * options.bias);
00217                     else
00218                         pqueue.push(*node, data[*node]);
00219                     break;
00220                 }
00221             }
00222         }
00223     }
00224     
00225     LabelType contourLabel = maxRegionLabel + 1;  // temporary contour label
00226     
00227     // perform region growing
00228     while(!pqueue.empty())
00229     {
00230         Node node = pqueue.top();
00231         CostType cost = pqueue.topPriority();
00232         pqueue.pop();
00233         
00234         if((options.terminate & StopAtThreshold) && (cost > options.max_cost))
00235             break;
00236 
00237         LabelType label = labels[node];
00238         
00239         if(label == contourLabel)
00240             continue;
00241 
00242         // Put the unlabeled neighbors in the priority queue.
00243         for (neighbor_iterator arc(g, node); arc != INVALID; ++arc)
00244         {
00245             LabelType neighborLabel = labels[g.target(*arc)];
00246             if(neighborLabel == 0)
00247             {
00248                 labels[g.target(*arc)] = label;
00249                 CostType priority = (label == options.biased_label)
00250                                        ? data[g.target(*arc)] * options.bias
00251                                        : data[g.target(*arc)];
00252                 if(priority < cost)
00253                     priority = cost;
00254                 pqueue.push(g.target(*arc), priority);
00255             }
00256             else if(keepContours && (label != neighborLabel) && (neighborLabel != contourLabel))
00257             {
00258                 // The present neighbor is adjacent to more than one region
00259                 // => mark it as contour.
00260                 CostType priority = (neighborLabel == options.biased_label)
00261                                        ? data[g.target(*arc)] * options.bias
00262                                        : data[g.target(*arc)];
00263                 if(cost < priority) // neighbor not yet processed
00264                     labels[g.target(*arc)] = contourLabel;
00265             }
00266         }
00267     }
00268     
00269     if(keepContours)
00270     {
00271         // Replace the temporary contour label with label 0.
00272         typename T2Map::iterator k   = labels.begin(),
00273                                  end = labels.end();
00274         for(; k != end; ++k)
00275             if(*k == contourLabel)
00276                 *k = 0;
00277     }
00278     
00279     return maxRegionLabel;
00280 }

00284 template <class Graph, class T1Map, class T2Map>
00285 typename T2Map::value_type 
00286 watershedsGraph(Graph const & g, 
00287                 T1Map const & data,
00288                 T2Map & labels,
00289                 WatershedOptions const & options)
00290 {
00291     typedef typename T2Map::value_type LabelType;
00292     
00293     if(options.method == WatershedOptions::UnionFind)
00294     {
00295         vigra_precondition(g.maxDegree() <= NumericTraits<unsigned short>::max(),
00296             "watershedsGraph(): cannot handle nodes with degree > 65535.");
00297             
00298         typename Graph::template NodeMap<unsigned short>  lowestNeighborIndex(g);
00299         
00300         graph_detail::prepareWatersheds(g, data, lowestNeighborIndex);
00301         return graph_detail::unionFindWatersheds(g, data, lowestNeighborIndex, labels);
00302     }
00303     else if(options.method == WatershedOptions::RegionGrowing)
00304     {
00305         SeedOptions seed_options;
00306         
00307         // check if the user has explicitly requested seed computation
00308         if(options.seed_options.mini != SeedOptions::Unspecified)
00309         {
00310             seed_options = options.seed_options;
00311         }
00312         else
00313         {
00314             // otherwise, don't compute seeds if 'labels' already contains them 
00315             if(labels.any())
00316                 seed_options.mini = SeedOptions::Unspecified;
00317         }
00318 
00319         if(seed_options.mini != SeedOptions::Unspecified)
00320         {
00321             graph_detail::generateWatershedSeeds(g, data, labels, seed_options);
00322         }
00323         
00324         return graph_detail::seededWatersheds(g, data, labels, options);
00325     }
00326     else
00327     {
00328         vigra_precondition(false,
00329            "watershedsGraph(): invalid method in watershed options.");
00330         return 0;
00331     }
00332 }

00354 /** \brief Watershed segmentation of an arbitrary-dimensional array.
00355 
00356     This function implements variants of the watershed algorithms
00357     described in
00358 
00359     [1] L. Vincent and P. Soille: <em>"Watersheds in digital spaces: An efficient algorithm
00360     based on immersion simulations"</em>, IEEE Trans. Patt. Analysis Mach. Intell. 13(6):583-598, 1991
00361 
00362     [2] J. Roerdink, R. Meijster: <em>"The watershed transform: definitions, algorithms,
00363     and parallelization strategies"</em>, Fundamenta Informaticae, 41:187-228, 2000
00364 
00365     The source array \a data is a boundary indicator such as the gaussianGradientMagnitude()
00366     or the trace of the \ref boundaryTensor(), and the destination \a labels is a label array
00367     designating membership of each point in one of the regions found. Plateaus in the boundary
00368     indicator are handled via simple tie breaking strategies. Argument \a neighborhood 
00369     specifies the connectivity between points and can be <tt>DirectNeighborhood</tt> (meaning 
00370     4-neighborhood in 2D and 6-neighborhood in 3D, default) or <tt>IndirectNeighborhood</tt> 
00371     (meaning 8-neighborhood in 2D and 26-neighborhood in 3D).
00372     
00373     The watershed variant to be applied can be selected in the \ref WatershedOptions
00374     object: When you call <tt>WatershedOptions::regionGrowing()</tt> (default), the flooding
00375     algorithm from [1] is used. Alternatively, <tt>WatershedOptions::unionFind()</tt> uses
00376     the scan-line algorithm 4.7 from [2]. The latter is faster, but does not support any options 
00377     (if you pass options nonetheless, they are silently ignored).
00378     
00379     The region growing algorithm needs a seed for each region. Seeds can either be provided in
00380     the destination array \a labels (which will then be overwritten with the result) or computed
00381     automatically by an internal call to generateWatershedSeeds(). In the former case you have 
00382     full control over seed placement, while the latter is more convenient. Automatic seed 
00383     computation is performed when you provide seeding options via <tt>WatershedOptions::seedOptions()</tt> 
00384     or when the array \a labels is empty (all zeros), in which case default seeding options 
00385     are chosen. The destination image should be zero-initialized for automatic seed computation.
00386     
00387     Further options to be specified via \ref WatershedOptions are:
00388     
00389     <ul>
00390     <li> <tt>keepContours()</tt>: Whether to keep a 1-pixel-wide contour (with label 0) between 
00391          regions (otherwise, a complete grow is performed, i.e. all pixels are assigned to a region).
00392     <li> <tt>stopAtThreshold()</tt>: Whether to stop growing when the boundaryness exceeds a threshold 
00393          (remaining pixels keep label 0).
00394     <li> <tt>biasLabel()</tt>: Whether one region (label) is to be preferred or discouraged by biasing its cost 
00395          with a given factor (smaller than 1 for preference, larger than 1 for discouragement).
00396     </ul>
00397     
00398     The option <tt>turboAlgorithm()</tt> is implied by method <tt>regionGrowing()</tt> (this is
00399     in contrast to watershedsRegionGrowing(), which supports an additional algorithm in 2D only).
00400 
00401     watershedsMultiArray() returns the number of regions found (= the highest region label, because 
00402     labels start at 1). 
00403 
00404     <b> Declaration:</b>
00405 
00406     \code
00407     namespace vigra {
00408         template <unsigned int N, class T, class S1,
00409                                   class Label, class S2>
00410         Label
00411         watershedsMultiArray(MultiArrayView<N, T, S1> const & data,
00412                              MultiArrayView<N, Label, S2> labels,  // may also hold input seeds
00413                              NeighborhoodType neighborhood = DirectNeighborhood,
00414                              WatershedOptions const & options = WatershedOptions());
00415     }
00416     \endcode
00417 
00418     <b> Usage:</b>
00419 
00420     <b>\#include</b> <vigra/multi_watersheds.hxx><br>
00421     Namespace: vigra
00422 
00423     Example: watersheds of the gradient magnitude (the example works likewise for higher dimensions).
00424 
00425     \code
00426     MultiArray<2, unsigned char> src(Shape2(w, h));
00427     ... // read input data
00428     
00429     // compute gradient magnitude at scale 1.0 as a boundary indicator
00430     MultiArray<2, float> gradMag(src.shape());
00431     gaussianGradientMagnitude(srcImageRange(src), destImage(gradMag), 1.0);
00432 
00433     // example 1
00434     {
00435         // the pixel type of the destination image must be large enough to hold
00436         // numbers up to 'max_region_label' to prevent overflow
00437         MultiArray<2, unsigned int> labeling(src.shape());
00438         
00439         // call region-growing algorithm for 4-neighborhood, leave a 1-pixel boundary between 
00440         // regions, and autogenerate seeds from all gradient minima where the magnitude is 
00441         // less than 2.0.
00442         unsigned int max_region_label = 
00443               watershedsMultiArray(gradMag, labeling, DirectNeighborhood,
00444                                    WatershedOptions().keepContours()
00445                                       .seedOptions(SeedOptions().minima().threshold(2.0)));
00446     }
00447     
00448     // example 2
00449     {
00450         MultiArray<2, unsigned int> labeling(src.shape());
00451         
00452         // compute seeds beforehand (use connected components of all pixels 
00453         // where the gradient is below 4.0)
00454         unsigned int max_region_label = generateWatershedSeeds(gradMag, labeling,
00455                                                        SeedOptions().levelSets(4.0));
00456         
00457         // quantize the gradient image to 256 gray levels
00458         float m, M;
00459         gradMag.minmax(&m, &M);
00460         
00461         using namespace multi_math;
00462         MultiArray<2, unsigned char> gradMag256(255.0 / (M - m) * (gradMag - m));
00463         
00464         // call region-growing algorithm with 8-neighborhood,
00465         // since the data are 8-bit, a faster priority queue will be used
00466         watershedsMultiArray(gradMag256, labeling, IndirectNeighborhood);
00467     }
00468     
00469     // example 3
00470     {
00471         MultiArray<2, unsigned int> labeling(src.shape());
00472         
00473         .. // put seeds in 'labeling', e.g. from an interactive labeling program,
00474            // make sure that label 1 corresponds to the background
00475         
00476         // bias the watershed algorithm so that the background is preferred
00477         // by reducing the cost for label 1 to 90%
00478         watershedsMultiArray(gradMag, labeling, 
00479                              WatershedOptions().biasLabel(1, 0.9));
00480     }
00481     
00482     // example 4
00483     {
00484         MultiArray<2, unsigned int> labeling(src.shape());
00485         
00486         // use the fast union-find algorithm with 4-neighborhood
00487         watershedsMultiArray(gradMag, labeling, WatershedOptions().unionFind());
00488     }
00489     \endcode
00490 */
