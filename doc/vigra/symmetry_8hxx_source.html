00058 /** \brief Find centers of radial symmetry in an image.
00059 
00060     This algorithm implements the Fast Radial Symmetry Transform according to
00061     [G. Loy, A. Zelinsky: <em> "A Fast Radial Symmetry Transform for Detecting
00062     Points of Interest"</em>, in: A. Heyden et al. (Eds.): Proc. of 7th European
00063     Conf. on Computer Vision, Part 1, pp. 358-368, Springer LNCS 2350, 2002].
00064     Minima of the algorithm response mark dark blobs, maxima correspond to light blobs.
00065     The "radial strictness parameter" is fixed at <TT>alpha</tt> = 2.0, the
00066     spatial spreading of the raw response is done by a Gaussian convolution
00067     at <tt>0.25*scale</TT> (these values are recommendations from the paper).
00068     Loy and Zelinsky additionally propose to add the operator response from several
00069     scales (see usage example below).
00070 
00071     <b> Declarations:</b>
00072 
00073     pass 2D array views:
00074     \code

00080                                 MultiArrayView<2, T2, S2> dest,
00081                                 double scale);
00082     }
00083     \endcode

00178 template <class SrcIterator, class SrcAccessor,
00179           class DestIterator, class DestAccessor>
00180 void
00181 radialSymmetryTransform(SrcIterator sul, SrcIterator slr, SrcAccessor as,
00182                DestIterator dul, DestAccessor ad,
00183         double scale)
00184 {
00185     vigra_precondition(scale > 0.0,
00186                  "radialSymmetryTransform(): Scale must be > 0");
00187 
00188     int w = slr.x - sul.x;
00189     int h = slr.y - sul.y;
00190 
00191     if(w <= 0 || h <= 0) return;
00192 
00193     typedef typename
00194         NumericTraits<typename SrcAccessor::value_type>::RealPromote TmpType;
00195 
00196     typedef BasicImage<TmpType> TmpImage;
00197     typedef typename TmpImage::Iterator TmpIterator;
00198 
00199     TmpImage gx(w,h);
00200     TmpImage gy(w,h);
00201     IImage   orientationCounter(w,h);
00202     TmpImage magnitudeAccumulator(w,h);
00203 
00204     gaussianGradient(srcIterRange(sul, slr, as),
00205                      destImage(gx), destImage(gy),
00206                      scale);
00207 
00208     orientationCounter.init(0);
00209     magnitudeAccumulator.init(NumericTraits<TmpType>::zero());
00210 
00211     TmpIterator gxi = gx.upperLeft();
00212     TmpIterator gyi = gy.upperLeft();
00213     int y;
00214     for(y=0; y<h; ++y, ++gxi.y, ++gyi.y)
00215     {
00216         typename TmpIterator::row_iterator gxr = gxi.rowIterator();
00217         typename TmpIterator::row_iterator gyr = gyi.rowIterator();
00218 
00219         for(int x = 0; x<w; ++x, ++gxr, ++gyr)
00220         {
00221             double angle = VIGRA_CSTD::atan2(-*gyr, *gxr);
00222             double magnitude = VIGRA_CSTD::sqrt(*gxr * *gxr + *gyr * *gyr);
00223 
00224             if(magnitude < NumericTraits<TmpType>::epsilon()*10.0)
00225                 continue;
00226 
00227             int dx = NumericTraits<int>::fromRealPromote(scale * VIGRA_CSTD::cos(angle));
00228             int dy = NumericTraits<int>::fromRealPromote(scale * VIGRA_CSTD::sin(angle));
00229 
00230             int xx = x + dx;
00231             int yy = y - dy;
00232 
00233             if(xx >= 0 && xx < w && yy >= 0 && yy < h)
00234             {
00235                 orientationCounter(xx, yy) += 1;
00236                 magnitudeAccumulator(xx, yy) += detail::RequiresExplicitCast<TmpType>::cast(magnitude);
00237             }
00238 
00239             xx = x - dx;
00240             yy = y + dy;
00241 
00242             if(xx >= 0 && xx < w && yy >= 0 && yy < h)
00243             {
00244                 orientationCounter(xx, yy) -= 1;
00245                 magnitudeAccumulator(xx, yy) -= detail::RequiresExplicitCast<TmpType>::cast(magnitude);
00246             }
00247         }
00248     }
00249 
00250     int maxOrientation = 0;
00251     TmpType maxMagnitude = NumericTraits<TmpType>::zero();
00252 
00253     for(y=0; y<h; ++y)
00254     {
00255         for(int x = 0; x<w; ++x)
00256         {
00257             int o = VIGRA_CSTD::abs(orientationCounter(x,y));
00258 
00259             if(o > maxOrientation)
00260                 maxOrientation = o;
00261 
00262             TmpType m = VIGRA_CSTD::abs(magnitudeAccumulator(x,y));
00263 
00264             if(m > maxMagnitude)
00265                 maxMagnitude = m;
00266         }
00267     }
00268 
00269     for(y=0; y<h; ++y)
00270     {
00271         for(int x = 0; x<w; ++x)
00272         {
00273             double o = (double)orientationCounter(x, y) / maxOrientation;
00274             magnitudeAccumulator(x, y) = detail::RequiresExplicitCast<TmpType>::cast(o * o * magnitudeAccumulator(x, y) / maxMagnitude);
00275         }
00276     }
00277 
00278     gaussianSmoothing(srcImageRange(magnitudeAccumulator), destIter(dul, ad), 0.25*scale);
00279 }
