00359 // A fake "pointer" for MultiIterator containing coordinates.
00360 // Indices (or a pointer) cannot be circumvented in coordiante iterators,
00361 // since floating point addition is not associative and
00362 // iterator comparison is done via via '<' or '!='. Class CoordinateStride
00363 // thus forwards iterator comparison to the index or pointer part
00364 // of its template parameter S.
00365 template<unsigned N, class S = StridePair<N> >
00366 class CoordinateStride : protected S
00367 {
00368   public:
00369     typedef          MultiArrayIndex     difference_type;
00370     typedef typename S::stride_type      stride_type;
00371     typedef typename S::deref_type       deref_type;
00372     typedef          CoordinateStride<N> type;
00373     typedef typename S::coord_type       coord_type;
00374     typedef typename S::index_type       index_type;
00375     typedef typename S::shape_array_type shape_array_type;
00376 
00377   protected:
00378     double stride_0;
00379 
00380     CoordinateStride(void*) {} // used MultiIterator ctor, unused.
00381 
00382   public:
00383     CoordinateStride(const S & x, double s0)
00384         : S(x), stride_0(s0) {}
00385 
00386 #ifndef DOXYGEN
00387     using S::operator*;
00388     using S::idx0;
00389     using S::idx;
00390     using S::dim0;
00391     using S::operator+=;
00392     using S::operator-=;
00393 #endif
00394 
00395     void operator++()
00396     {
00397         ++idx0();
00398         dim0() += stride_0;
00399     }
00400     void operator--()
00401     {
00402         --idx0();
00403         dim0() -= stride_0;
00404     }
00405     void operator+=(difference_type n)
00406     {
00407         idx0() += n;
00408         dim0() += n * stride_0;
00409     }
00410     void operator-=(difference_type n)
00411     {
00412         idx0() -= n;
00413         dim0() -= n * stride_0;
00414     }
00415 
00416     stride_type operator[](difference_type n) const
00417     {
00418         type ret = *this;
00419         ret[0] += n;
00420         return ret;
00421     }
00422 
00423     stride_type operator[](stride_type x) const
00424     {
00425         return *this + x;
00426     }
00427 
00428     // ... but use the idx() for comparisons:
00429     bool operator!=(const CoordinateStride & y) const
00430     {
00431         return idx() != y.idx();
00432     }
00433     bool operator==(const CoordinateStride & y) const
00434     {
00435         if (stride_0 != y.stride_0)
00436             return false;
00437         return idx() == y.idx();
00438     }
00439     bool operator<(const CoordinateStride & y) const
00440     {
00441         return idx() < y.idx();
00442     }
00443 
00444     bool operator<=(const CoordinateStride & y) const
00445     {
00446         if (stride_0 == y.stride_0)
00447             return true;
00448         return *this < y;
00449     }
00450     bool operator>(const CoordinateStride & y) const
00451     {
00452         return y < *this;
00453     }
00454     bool operator>=(const CoordinateStride & y) const
00455     {
00456         if (stride_0 == y.stride_0)
00457             return true;
00458         return operator>(y);
00459     }

00461     friend std::ostream &
00462     operator<<(std::ostream & os, const CoordinateStride & x)
00463     {
00464         os << "{" << x.stride_0 << ": " << static_cast<const S &>(x) << "}";
00465         return os;
00466     }
