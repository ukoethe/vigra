00619 MultiArrayView.  By default, MultiArrayViews are tagged as
00620 unstrided. If necessary, strided arrays are constructed automatically
00621 by calls to a variant of the bind...() function.
00622 
00623 In addition to the member functions described here, <tt>MultiArrayView</tt>
00624 and its subclasses support arithmetic and algebraic functions via the 
00625 module \ref MultiMathModule.
00626

00628 <tt>MultiArrayView</tt> of appropriate (2-dimensional) shape, you can
00629 create a \ref vigra::BasicImageView that acts as a wrapper with the
00630 necessary interface -- see \ref MultiArrayToImage.
00631

00633 \code
00634     N: the array dimension
00635 
00636     T: the type of the array elements
00637

00639        or not. An array is unstrided if the array elements occupy consecutive
00640        memory location, strided if there is an offset in between (e.g.
00641        when a view is created that skips every other array element).
00642        The compiler can generate faster code for unstrided arrays.
00643        Possible values: UnstridedArrayTag (default), StridedArrayTag

01907 template <unsigned int N, class T, class StrideTag>
01908 template <class CN>
01909 bool
01910 MultiArrayView <N, T, StrideTag>::arraysOverlap(const MultiArrayView <N, T, CN>& rhs) const
01911 {
01912     vigra_precondition (shape () == rhs.shape (),
01913         "MultiArrayView::arraysOverlap(): shape mismatch.");
01914     const_pointer first_element = this->m_ptr,
01915                   last_element = first_element + dot(this->m_shape - difference_type(1), this->m_stride);
01916     typename MultiArrayView <N, T, CN>::const_pointer
01917            rhs_first_element = rhs.data(),
01918            rhs_last_element = rhs_first_element + dot(rhs.shape() - difference_type(1), rhs.stride());
01919     return !(last_element < rhs_first_element || rhs_last_element < first_element);

02015 {
02016     difference_type permutation;
02017     for(int k=0; k<(int)N; ++k)
02018         permutation[k] = k;
