00512 template <class SrcImageIterator, class SrcAccessor,
00513           class DestImageIterator, class DestAccessor, class Functor>
00514 void
00515 gradientBasedTransform(SrcImageIterator srcul, SrcImageIterator srclr, SrcAccessor sa,
00516                        DestImageIterator destul, DestAccessor da, Functor const & grad)
00517 {
00518     int w = srclr.x - srcul.x;
00519     int h = srclr.y - srcul.y;
00520     int x,y;
00521 
00522     SrcImageIterator sy = srcul;
00523     DestImageIterator dy = destul;
00524 
00525     const Diff2D left(-1,0);
00526     const Diff2D right(1,0);
00527     const Diff2D top(0,-1);
00528     const Diff2D bottom(0,1);
00529 
00530     typedef typename NumericTraits<typename SrcAccessor::value_type>::RealPromote TmpType;
00531     TmpType diffx, diffy;
00532 
00533     SrcImageIterator sx = sy;
00534     DestImageIterator dx = dy;
00535 
00536     diffx = sa(sx) - sa(sx, right);
00537     diffy = sa(sx) - sa(sx, bottom);
00538     da.set(grad(diffx, diffy), dx);
00539 
00540     for(x=2, ++sx.x, ++dx.x; x<w; ++x, ++sx.x, ++dx.x)
00541     {
00542         diffx = (sa(sx, left) - sa(sx, right)) / TmpType(2.0);
00543         diffy = sa(sx) - sa(sx, bottom);
00544         da.set(grad(diffx, diffy), dx);
00545     }
00546 
00547     diffx = sa(sx, left) - sa(sx);
00548     diffy = sa(sx) - sa(sx, bottom);
00549     da.set(grad(diffx, diffy), dx);
00550 
00551     ++sy.y;
00552     ++dy.y;
00553 
00554     for(y=2; y<h; ++y, ++sy.y, ++dy.y)
00555     {
00556         sx = sy;
00557         dx = dy;
00558 
00559         diffx = sa(sx) - sa(sx, right);
00560         diffy = (sa(sx, top) - sa(sx, bottom)) / TmpType(2.0);
00561         da.set(grad(diffx, diffy), dx);
00562 
00563         for(x=2, ++sx.x, ++dx.x; x<w; ++x, ++sx.x, ++dx.x)
00564         {
00565             diffx = (sa(sx, left) - sa(sx, right)) / TmpType(2.0);
00566             diffy = (sa(sx, top) - sa(sx, bottom)) / TmpType(2.0);
00567             da.set(grad(diffx, diffy), dx);
00568         }
00569 
00570         diffx = sa(sx, left) - sa(sx);
00571         diffy = (sa(sx, top) - sa(sx, bottom)) / TmpType(2.0);
00572         da.set(grad(diffx, diffy), dx);
00573     }
00574 
00575     sx = sy;
00576     dx = dy;
00577 
00578     diffx = sa(sx) - sa(sx, right);
00579     diffy = sa(sx, top) - sa(sx);
00580     da.set(grad(diffx, diffy), dx);
00581 
00582     for(x=2, ++sx.x, ++dx.x; x<w; ++x, ++sx.x, ++dx.x)
00583     {
00584         diffx = (sa(sx, left) - sa(sx, right)) / TmpType(2.0);
00585         diffy = sa(sx, top) - sa(sx);
00586         da.set(grad(diffx, diffy), dx);
00587     }
00588 
00589     diffx = sa(sx, left) - sa(sx);
00590     diffy = sa(sx, top) - sa(sx);
00591     da.set(grad(diffx, diffy), dx);

01027 /** \brief Adjust brightness and contrast of an image.
01028 
01029     This functor applies a gamma correction to each pixel in order to
01030     modify the brightness of the image. To the result of the gamma
01031     correction, another transform is applied that modifies the
01032     contrast. The brightness and contrast parameters must be
01033     positive. Values greater than 1 will increase image brightness or
01034     contrast respectively, values smaller than 1 decrease them.  A
01035     value of exactly 1 will have no effect.  If contrast is set to 1,
01036     the result is equivalent to that of the GammaFunctor with gamma =
01037     1./brightness.
01038 
01039     For \ref RGBValue "RGBValue's", the transforms are applied
01040     component-wise. The pixel values are assumed to lie between the
01041     given minimum and maximum values (in case of RGB, this is again
01042     understood component-wise). In case of <TT>unsigned char</TT>, min
01043     and max default to 0 and 255 respectively.  Precisely, the
01044     following transform is applied to each <em> PixelValue</em>:
01045 
01046     \f[
01047     \begin{array}{rcl}
01048     V_1 & = & \frac{PixelValue - min}{max - min} \\
01049     V_2 & = & V_1^\frac{1}{brightness} \\
01050     V_3 & = & 2 V_2 - 1 \\
01051     V_4 & = & \left\lbrace
01052         \begin{array}{l}
01053          V_3^\frac{1}{contrast} \mbox{\rm \quad if  } V_3 \ge 0 \\
01054          - (-V_3)^\frac{1}{contrast} \mbox{\rm \quad otherwise}
01055         \end{array} \right. \\
01056     Result & = & \frac{V_4 + 1}{2} (max - min) + min
01057     \end{array}
01058     \f]
01059 
01060     If the <TT>PixelType</TT> is <TT>unsigned char</TT>, a look-up-table is used
01061     for faster computation.
01062 
01063     <b> Traits defined:</b>
01064 
01065     <tt>FunctorTraits::isUnaryFunctor</tt> is true (<tt>VigraTrueType</tt>)
01066 
01067     <b> Usage:</b>
01068 
01069     <b>\#include</b> <vigra/transformimage.hxx><br>
01070     Namespace: vigra
01071 
01072     \code
01073     vigra::BImage bimage(width, height);
01074     double brightness, contrast;
01075     ...
01076     vigra::transformImage(srcImageRange(bimage), destImage(bimage),

01078 
01079 
01080 
01081     vigra::FImage fimage(width, height);
01082     ...
01083 
01084     vigra::FindMinMax<float> minmax;
01085     vigra::inspectImage(srcImageRange(fimage), minmax);
01086 
01087     vigra::transformImage(srcImageRange(fimage), destImage(fimage),
01088        vigra::BrightnessContrastFunctor<float>(brightness, contrast, minmax.min, minmax.max));
01089 
01090 
01091     \endcode
01092 
01093     <b> Required Interface:</b>
01094 
01095     Scalar types: must be a linear algebra (+, - *, NumericTraits),
01096     strict weakly ordered (<), and <TT>pow()</TT> must be defined.
01097 
01098     RGB values: the component type must meet the above requirements.
01099 */

01101 class BrightnessContrastFunctor
01102 {
01103     typedef typename
01104         NumericTraits<PixelType>::RealPromote promote_type;
01105 
01106  public:
01107 
01108         /** the functor's argument type
01109         */
01110     typedef PixelType argument_type;
01111 
01112         /** the functor's result type
01113         */
01114     typedef PixelType result_type;
01115 
01116         /** \deprecated use argument_type and result_type
01117         */
01118     typedef PixelType value_type;
01119 
01120         /** Init functor for argument range <TT>[min, max]</TT>.
01121             <TT>brightness</TT> and <TT>contrast</TT> values > 1 will
01122             increase brightness and contrast, < 1 will decrease them, and == 1 means
01123             no change.
01124         */
01125     BrightnessContrastFunctor(promote_type brightness, promote_type contrast,
01126                               argument_type const & min, argument_type const & max)
01127     : b_(1.0/brightness),
01128       c_(1.0/contrast),
01129       min_(min),
01130       diff_(max - min),
01131       zero_(NumericTraits<promote_type>::zero()),
01132       one_(NumericTraits<promote_type>::one())
01133     {}
01134 
01135         /** Calculate modified gray or color value
01136         */
01137     result_type operator()(argument_type const & v) const
01138     {
01139         promote_type v1 = (v - min_) / diff_;
01140         promote_type brighter = VIGRA_CSTD::pow(v1, b_);
01141         promote_type v2 = 2.0 * brighter - one_;

01143                                      -VIGRA_CSTD::pow(-v2, c_) :
01144                                       VIGRA_CSTD::pow(v2, c_);
01145         return result_type(0.5 * diff_ * (contrasted + one_) + min_);
01146     }
01147 
01148   private:
01149     promote_type b_, c_;
01150     argument_type min_;
01151     promote_type diff_, zero_, one_;
01152 };

01308     BrightnessContrastFunctor, which additionally changes the

01316     Precisely, the following transform is applied to each <em>
01317     PixelValue</em>:

01321     V_1 & = & \frac{PixelValue - min}{max - min} \\
01322     V_2 & = & V_1^{gamma} \\

01324     \end{array}

01328     look-up-table is used for faster computation.
