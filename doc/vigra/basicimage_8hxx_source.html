00057 template <class IMAGEITERATOR>
00058 class LineBasedColumnIteratorPolicy
00059 {
00060   public:
00061     typedef IMAGEITERATOR                             ImageIterator;
00062     typedef typename IMAGEITERATOR::LineStartIterator LineStartIterator;
00063     typedef typename IMAGEITERATOR::value_type        value_type;
00064     typedef typename IMAGEITERATOR::difference_type::MoveY
00065                                                       difference_type;
00066     typedef typename IMAGEITERATOR::reference         reference;
00067     typedef typename IMAGEITERATOR::index_reference   index_reference;
00068     typedef typename IMAGEITERATOR::pointer           pointer;
00069     typedef std::random_access_iterator_tag           iterator_category;
00070 
00071 
00072     struct BaseType
00073     {
00074         explicit BaseType(LineStartIterator c = LineStartIterator(),
00075                           difference_type o = 0)
00076         : line_start_(c), offset_(o)
00077         {}
00078 
00079         LineStartIterator line_start_;
00080         difference_type offset_;
00081     };
00082 
00083     static void initialize(BaseType &) {}
00084 
00085     static reference dereference(BaseType const & d)
00086         { return const_cast<reference>(*(*d.line_start_ + d.offset_)); }
00087 
00088     static index_reference dereference(BaseType const & d, difference_type n)
00089     {
00090         return const_cast<index_reference>(*(d.line_start_[n] + d.offset_));
00091     }
00092 
00093     static bool equal(BaseType const & d1, BaseType const & d2)
00094         { return d1.line_start_ == d2.line_start_; }
00095 
00096     static bool less(BaseType const & d1, BaseType const & d2)
00097         { return d1.line_start_ < d2.line_start_; }
00098 
00099     static difference_type difference(BaseType const & d1, BaseType const & d2)
00100         { return d1.line_start_ - d2.line_start_; }
00101 
00102     static void increment(BaseType & d)
00103         { ++d.line_start_; }
00104 
00105     static void decrement(BaseType & d)
00106         { --d.line_start_; }
00107 
00108     static void advance(BaseType & d, difference_type n)
00109         { d.line_start_ += n; }
00110 };

01138 template <class PIXELTYPE, class Alloc>
01139 void
01140 BasicImage<PIXELTYPE, Alloc>::resizeImpl(int width, int height, value_type const & d, bool skipInit)
01141 {
01142     vigra_precondition((width >= 0) && (height >= 0),
01143          "BasicImage::resize(int width, int height, value_type const &): "
01144          "width and height must be >= 0.\n");
01145     vigra_precondition(width * height >= 0,
01146          "BasicImage::resize(int width, int height, value_type const &): "
01147          "width * height too large (integer overflow -> negative).\n");
01148 
01149     if (width_ != width || height_ != height)  // change size?
01150     {
01151         value_type * newdata = 0;
01152         value_type ** newlines = 0;
01153         if(width*height > 0)
01154         {
01155             if (width*height != width_*height_) // different sizes, must reallocate
01156             {
01157                 newdata = allocator_.allocate(typename Alloc::size_type(width*height));
01158                 if(!skipInit)
01159                     std::uninitialized_fill_n(newdata, width*height, d);
01160                 newlines = initLineStartArray(newdata, width, height);
01161                 deallocate();
01162             }
01163             else // need only to reshape
01164             {
01165                 newdata = data_;
01166                 if(!skipInit)
01167                     std::fill_n(newdata, width*height, d);
01168                 newlines = initLineStartArray(newdata, width, height);
01169                 pallocator_.deallocate(lines_, typename Alloc::size_type(height_));
01170             }
01171         }
01172         else
01173         {
01174             deallocate();
01175         }
01176 
01177         data_ = newdata;
01178         lines_ = newlines;
01179         width_ = width;
01180         height_ = height;
01181     }
01182     else if(width*height > 0 && !skipInit) // keep size, re-init data
01183     {
01184         std::fill_n(data_, width*height, d);
01185     }
01186 }
