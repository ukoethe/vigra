00175        of the image, and white is on the left.

00177 template <class SrcIterator, class SrcAccessor, class DestImage>
00178 unsigned int
00179 prepareSlantedEdgeInput(SrcIterator sul, SrcIterator slr, SrcAccessor src, DestImage & res,
00180                         SlantedEdgeMTFOptions const & options)
00181 {
00182     unsigned int w = slr.x - sul.x;
00183     unsigned int h = slr.y - sul.y;
00184 
00185     // find rough estimate of the edge
00186     ArrayVector<Edgel> edgels;
00187     cannyEdgelList(sul, slr, src, edgels, 2.0);
00188     std::sort(edgels.begin(), edgels.end(), SortEdgelsByStrength());
00189 
00190     double x = 0.0, y = 0.0, x2 = 0.0, y2 = 0.0, xy = 0.0;
00191     unsigned int c = std::min(w, h);
00192 
00193     for(unsigned int k = 0; k < c; ++k)
00194     {
00195         x += edgels[k].x;
00196         y += edgels[k].y;
00197         x2 += sq(edgels[k].x);
00198         xy += edgels[k].x*edgels[k].y;
00199         y2 += sq(edgels[k].y);
00200     }
00201     double xc = x / c, yc = y / c;
00202     x2 = x2 / c - sq(xc);
00203     xy = xy / c - xc*yc;
00204     y2 = y2 / c - sq(yc);
00205     double angle = 0.5*VIGRA_CSTD::atan2(2*xy, x2 - y2);
00206 
00207     DestImage tmp;
00208     // rotate image when slope is less than +-45 degrees
00209     if(VIGRA_CSTD::fabs(angle) < M_PI / 4.0)
00210     {
00211         xc = yc;
00212         yc = w - xc - 1;
00213         std::swap(w, h);
00214         tmp.resize(w, h);
00215         rotateImage(srcIterRange(sul, slr, src), destImage(tmp), 90);
00216         angle += M_PI / 2.0;
00217     }
00218     else
00219     {
00220         tmp.resize(w, h);
00221         copyImage(srcIterRange(sul, slr, src), destImage(tmp));
00222         if(angle < 0.0)
00223             angle += M_PI;
00224     }
00225     // angle is now between pi/4 and 3*pi/4
00226     double slope = VIGRA_CSTD::tan(M_PI/2.0 - angle);
00227     vigra_precondition(slope != 0.0,
00228           "slantedEdgeMTF(): Input edge is not slanted");
00229 
00230     // trim image so that the edge only intersects the top and bottom border
00231     unsigned int minimumNumberOfLines = options.minimum_number_of_lines, //20,
00232                  edgeWidth = options.desired_edge_width, // 10
00233                  minimumEdgeWidth = options.minimum_edge_width; // 5
00234 
00235     int y0 = 0, y1 = h;
00236     for(; edgeWidth >= minimumEdgeWidth; --edgeWidth)
00237     {
00238         y0 = int(VIGRA_CSTD::floor((edgeWidth - xc) / slope + yc + 0.5));
00239         y1 = int(VIGRA_CSTD::floor((w - edgeWidth - 1 - xc) / slope + yc + 0.5));
00240         if(slope < 0.0)
00241             std::swap(y0, y1);
00242         if(y1 - y0 >= (int)minimumNumberOfLines)
00243             break;
00244     }
00245 
00246     vigra_precondition(edgeWidth >= minimumEdgeWidth,
00247         "slantedEdgeMTF(): Input edge is too slanted or image is too small");
00248 
00249     y0 = std::max(y0, 0);
00250     y1 = std::min(y1+1, (int)h);
00251 
00252     res.resize(w, y1-y0);
00253 
00254     // ensure that white is on the left
00255     if(tmp(0,0) < tmp(w-1, h-1))
00256     {
00257         rotateImage(srcIterRange(tmp.upperLeft() + Diff2D(0, y0), tmp.upperLeft() + Diff2D(w, y1), tmp.accessor()),
00258                     destImage(res), 180);
00259     }
00260     else
00261     {
00262         copyImage(srcImageRange(tmp), destImage(res));
00263     }
00264     return edgeWidth;
00265 }

00267 template <class Image>
00268 void slantedEdgeShadingCorrection(Image & i, unsigned int edgeWidth)
00269 {
00270     using namespace functor;
00271 
00272     // after prepareSlantedEdgeInput(), the white region is on the left
00273     // find a plane that approximates the logarithm of the white ROI
00274 
00275     transformImage(srcImageRange(i), destImage(i), log(Arg1() + Param(1.0)));
00276 
00277     unsigned int w = i.width(),
00278                  h = i.height();
00279 
00280     Matrix<double> m(3,3), r(3, 1), l(3, 1);
00281     for(unsigned int y = 0; y < h; ++y)
00282     {
00283         for(unsigned int x = 0; x < edgeWidth; ++x)
00284         {
00285             l(0,0) = x;
00286             l(1,0) = y;
00287             l(2,0) = 1.0;
00288             m += outer(l);
00289             r += i(x,y)*l;
00290         }
00291     }
00292 
00293     linearSolve(m, r, l);
00294     double a = l(0,0),
00295            b = l(1,0),
00296            c = l(2,0);
00297 
00298     // subtract the plane and go back to the non-logarithmic representation
00299     for(unsigned int y = 0; y < h; ++y)
00300     {
00301         for(unsigned int x = 0; x < w; ++x)
00302         {
00303             i(x, y) = VIGRA_CSTD::exp(i(x,y) - a*x - b*y - c);
00304         }
00305     }
00306 }

00308 template <class Image, class BackInsertable>
00309 void slantedEdgeSubpixelShift(Image const & img, BackInsertable & centers, double & angle,
00310                               SlantedEdgeMTFOptions const & options)
00311 {
00312     unsigned int w = img.width();
00313     unsigned int h = img.height();
00314 
00315     Image grad(w, h);
00316     Kernel1D<double> kgrad;
00317     kgrad.initGaussianDerivative(1.0, 1);
00318 
00319     separableConvolveX(srcImageRange(img), destImage(grad), kernel1d(kgrad));
00320 
00321     int desiredEdgeWidth = (int)options.desired_edge_width;
00322     double sy = 0.0, sx = 0.0, syy = 0.0, sxy = 0.0;
00323     for(unsigned int y = 0; y < h; ++y)
00324     {
00325         double a = 0.0,
00326                b = 0.0;
00327         for(unsigned int x = 0; x < w; ++x)
00328         {
00329             a += x*grad(x,y);
00330             b += grad(x,y);
00331         }
00332         int c = int(a / b);
00333         // c is biased because the numbers of black and white pixels differ
00334         // repeat the analysis with a symmetric window around the edge
00335         a = 0.0;
00336         b = 0.0;
00337         int ew = desiredEdgeWidth;
00338         if(c-desiredEdgeWidth < 0)
00339             ew = c;
00340         if(c + ew + 1 >= (int)w)
00341             ew = w - c - 1;
00342         for(int x = c-ew; x < c+ew+1; ++x)
00343         {
00344             a += x*grad(x,y);
00345             b += grad(x,y);
00346         }
00347         double sc = a / b;
00348         sy += y;
00349         sx += sc;
00350         syy += sq(y);
00351         sxy += sc*y;
00352     }
00353     // fit a line to the subpixel locations
00354     double a = (h * sxy - sx * sy) / (h * syy - sq(sy));
00355     double b = (sx * syy - sxy * sy) / (h * syy - sq(sy));
00356 
00357     // compute the regularized subpixel values of the edge location
00358     angle = VIGRA_CSTD::atan(a);
00359     for(unsigned int y = 0; y < h; ++y)
00360     {
00361         centers.push_back(a * y + b);
00362     }
00363 }

00365 template <class Image, class Vector>
00366 void slantedEdgeCreateOversampledLine(Image const & img, Vector const & centers,
00367                                       Image & result)
00368 {
00369     unsigned int w = img.width();
00370     unsigned int h = img.height();
00371     unsigned int w2 = std::min(std::min(int(centers[0]), int(centers[h-1])),
00372                                std::min(int(w - centers[0]) - 1, int(w - centers[h-1]) - 1));
00373     unsigned int ww = 8*w2;
00374 
00375     Image r(ww, 1), s(ww, 1);
00376     for(unsigned int y = 0; y < h; ++y)
00377     {
00378         int x0 = int(centers[y]) - w2;
00379         int x1 = int((VIGRA_CSTD::ceil(centers[y]) - centers[y])*4);
00380         for(; x1 < (int)ww; x1 += 4)
00381         {
00382             r(x1, 0) += img(x0, y);
00383             ++s(x1, 0);
00384             ++x0;
00385         }
00386     }
00387 
00388     for(unsigned int x = 0; x < ww; ++x)
00389     {
00390         vigra_precondition(s(x,0) > 0.0,
00391            "slantedEdgeMTF(): Input edge is not slanted enough");
00392         r(x,0) /= s(x,0);
00393     }
00394 
00395     result.resize(ww-1, 1);
00396     for(unsigned int x = 0; x < ww-1; ++x)
00397     {
00398         result(x,0) = r(x+1,0) - r(x,0);
00399     }
00400 }

00402 template <class Image, class BackInsertable>
00403 void slantedEdgeMTFImpl(Image const & i, BackInsertable & mtf, double angle,
00404                         SlantedEdgeMTFOptions const & options)
00405 {
00406     unsigned int w = i.width();
00407     unsigned int h = i.height();
00408 
00409     double slantCorrection = VIGRA_CSTD::cos(angle);
00410     int desiredEdgeWidth = 4*options.desired_edge_width;
00411 
00412     Image magnitude;
00413 
00414     if(w - 2*desiredEdgeWidth < 64)
00415     {
00416         FFTWComplexImage otf(w, h);
00417         fourierTransform(srcImageRange(i), destImage(otf));
00418 
00419         magnitude.resize(w, h);
00420         for(unsigned int y = 0; y < h; ++y)
00421         {
00422             for(unsigned int x = 0; x < w; ++x)
00423             {
00424                 magnitude(x, y) = norm(otf(x, y));
00425             }
00426         }
00427     }
00428     else
00429     {
00430         w -= 2*desiredEdgeWidth;
00431         FFTWComplexImage otf(w, h);
00432         fourierTransform(srcImageRange(i, Rect2D(Point2D(desiredEdgeWidth, 0), Size2D(w, h))),
00433                          destImage(otf));
00434 
00435         // create an image where the edge is skipped - presumably it only contains the
00436         // noise which can then be subtracted
00437         Image noise(w,h);
00438         copyImage(srcImageRange(i, Rect2D(Point2D(0,0), Size2D(w/2, h))),
00439                   destImage(noise));
00440         copyImage(srcImageRange(i, Rect2D(Point2D(i.width()-w/2, 0), Size2D(w/2, h))),
00441                   destImage(noise, Point2D(w/2, 0)));
00442         FFTWComplexImage fnoise(w, h);
00443         fourierTransform(srcImageRange(noise), destImage(fnoise));
00444 
00445         // subtract the noise power spectrum from the total power spectrum
00446         magnitude.resize(w, h);
00447         for(unsigned int y = 0; y < h; ++y)
00448         {
00449             for(unsigned int x = 0; x < w; ++x)
00450             {
00451                 magnitude(x, y) = VIGRA_CSTD::sqrt(std::max(0.0, squaredNorm(otf(x, y))-squaredNorm(fnoise(x, y))));
00452             }
00453         }
00454     }
00455 
00456     Kernel1D<double> gauss;
00457     gauss.initGaussian(options.mtf_smoothing_scale);
00458     Image smooth(w,h);
00459     separableConvolveX(srcImageRange(magnitude), destImage(smooth), kernel1d(gauss));
00460 
00461     unsigned int ww = w/4;
00462     double maxVal = smooth(0,0),
00463            minVal = maxVal;
00464     for(unsigned int k = 1; k < ww; ++k)
00465     {
00466         if(smooth(k,0) >= 0.0 && smooth(k,0) < minVal)
00467             minVal = smooth(k,0);
00468     }
00469     double norm = maxVal-minVal;
00470 
00471     typedef typename BackInsertable::value_type Result;
00472     mtf.push_back(Result(0.0, 1.0));
00473     for(unsigned int k = 1; k < ww; ++k)
00474     {
00475         double n = (smooth(k,0) - minVal)/norm*sq(M_PI*k/w/VIGRA_CSTD::sin(M_PI*k/w));
00476         double xx = 4.0*k/w/slantCorrection;
00477         if(n < 0.0 || xx > 1.0)
00478             break;
00479         mtf.push_back(Result(xx, n));
00480     }
00481 }

00496 /** \brief Determine the magnitude transfer function of the camera.
00497 
00498     This operator estimates the magnitude transfer function (MTF) of a camera by means of the 
00499     slanted edge method described in:
00500     
00501     ISO Standard No. 12233: <i>"Photography - Electronic still picture cameras - Resolution measurements"</i>, 2000
00502     
00503     The input must be an image that contains a single step edge with bright pixels on one side and dark pixels on 
00504     the other. However, the intensity values must be neither saturated nor zero. The algorithms computes the MTF
00505     from the Fourier transform of the edge's derivative. Thus, if the actual MTF is anisotropic, the estimated 
00506     MTF does actually only apply in the direction perpendicular to the edge - several edges at different 
00507     orientations are required to estimate an anisotropic MTF.
00508     
00509     The algorithm returns a sequence of frequency / attenuation pairs. The frequency axis is normalized so that the
00510     Nyquist frequency of the original image is 0.5. Since the edge's derivative is computed with subpixel accuracy,
00511     the attenuation can usually be computed for frequencies significantly above the Nyquist frequency as well. The 
00512     MTF estimate ends at either the first zero crossing of the MTF or at frequency 1, whichever comes earlier.
00513     
00514     The present implementation improves the original slanted edge algorithm according to ISO 12233 in a number of
00515     ways:
00516     
00517     <ul>
00518     <li> The edge is not required to run nearly vertically or horizontally (i.e. with a slant of approximately 5 degrees).
00519          The algorithm will automatically compute the edge's actual angle and adjust estimates accordingly. 
00520          However, it is still necessary for the edge to be somewhat slanted, because subpixel-accurate estimation 
00521          of the derivative is impossible otherwise (i.e. the edge position perpendicular to the edge direction must 
00522          differ by at least 1 pixel between the two ends of the edge). 
00523          
00524     <li> Our implementation uses a more accurate subpixel derivative algorithm. In addition, we first perform a shading 
00525          correction in order to reduce possible derivative bias due to nonuniform illumination.
00526 
00527     <li> If the input image is large enough (i.e. there are at least 20 pixels on either side of the edge over
00528          the edge's entire length), our algorithm attempts to subtract the estimated noise power spectrum
00529          from the estimated MTF.
00530     </ul>
00531     
00532     The source value type <TT>T1</TT> must be a scalar type which is convertible to <tt>double</tt>.
00533     The result is written into the \a result sequence, which must be back-insertable (supports <tt>push_back()</tt>)
00534     and whose <tt>value_type</tt> must be constructible 
00535     from two <tt>double</tt> values. Algorithm options can be set via the \a options object 
00536     (see \ref vigra::NoiseNormalizationOptions for details).
00537     
00538     <b> Declarations:</b>
00539     
00540     pass 2D array views:
00541     \code
00542     namespace vigra {
00543         template <class T1, class S1, class BackInsertable>
00544         void
00545         slantedEdgeMTF(MultiArrayView<2, T1, S1> const & src, BackInsertable & mtf,
00546                        SlantedEdgeMTFOptions const & options = SlantedEdgeMTFOptions());
00547     }
00548     \endcode
00549     
00550     \deprecatedAPI{slantedEdgeMTF}
00551     pass \ref ImageIterators and \ref DataAccessors :
00552     \code
00553     namespace vigra {
00554         template <class SrcIterator, class SrcAccessor, class BackInsertable>
00555         void
00556         slantedEdgeMTF(SrcIterator sul, SrcIterator slr, SrcAccessor src, BackInsertable & mtf,
00557                        SlantedEdgeMTFOptions const & options = SlantedEdgeMTFOptions());
00558     }
00559     \endcode
00560     use argument objects in conjunction with \ref ArgumentObjectFactories :
00561     \code
00562     namespace vigra {
00563         template <class SrcIterator, class SrcAccessor, class BackInsertable>
00564         void
00565         slantedEdgeMTF(triple<SrcIterator, SrcIterator, SrcAccessor> src, BackInsertable & mtf,
00566                        SlantedEdgeMTFOptions const & options = SlantedEdgeMTFOptions())
00567     }
00568     \endcode
00569     \deprecatedEnd
00570     
00571     <b> Usage:</b>
00572     
00573     <b>\#include</b> <vigra/slanted_edge_mtf.hxx><br>
00574     Namespace: vigra
00575 
00576     \code

00579

00662 /** \brief Fit a Gaussian function to a given MTF.
00663 
00664     This function expects a sequence of frequency / attenuation pairs as produced by \ref slantedEdgeMTF()
00665     and finds the best fitting Gaussian point spread function (Gaussian functions are good approximations 
00666     of the PSF of many real cameras). It returns the standard deviation (scale) of this function. The algorithm
00667     computes the standard deviation by means of a linear least square on the logarithm of the MTF, i.e.
00668     an algebraic fit rather than a Euclidean fit - thus, the resulting Gaussian may not be the one that 
00669     intuitively fits the data optimally.
00670     
00671     <b> Declaration:</b>
00672     
00673     \code
00674     namespace vigra {
00675         template <class Vector>
00676         double mtfFitGaussian(Vector const & mtf);
00677     }
00678     \endcode
00679     
00680     <b> Usage:</b>
00681     
00682     <b>\#include</b> <vigra/slanted_edge_mtf.hxx><br>
00683     Namespace: vigra
00684     
00685     \code
00686     MultiArray<2, float> src(w,h);
00687     std::vector<vigra::TinyVector<double, 2> > mtf;
00688     
00689     ...
00690     slantedEdgeMTF(src, mtf);
00691     double scale = vigra::mtfFitGaussian(mtf)
00692     
00693     std::cout << "The camera PSF is approximately a Gaussian at scale " << scale << std::endl;
00694     \endcode
00695 
00696     <b> Required Interface:</b>
00697     
00698     \code
00699     Vector mtf;
00700     int numberOfMeasurements = mtf.size()
00701     
00702     double frequency = mtf[0][0];
00703     double attenuation = mtf[0][1];
00704     \endcode
00705 */
00706 template <class Vector>
00707 double mtfFitGaussian(Vector const & mtf)
00708 {
00709     double minVal = mtf[0][1];
00710     for(unsigned int k = 1; k < mtf.size(); ++k)
00711     {
00712         if(mtf[k][1] < minVal)
00713             minVal = mtf[k][1];
00714     }
00715     double x2 = 0.0,
00716            xy = 0.0;
00717     for(unsigned int k = 1; k < mtf.size(); ++k)
00718     {
00719         if(mtf[k][1] <= 0.0)
00720             break;
00721         double x = mtf[k][0],
00722                y = VIGRA_CSTD::sqrt(-VIGRA_CSTD::log(mtf[k][1])/2.0)/M_PI;
00723         x2 += vigra::sq(x);
00724         xy += x*y;
00725         if(mtf[k][1] == minVal)
00726             break;
00727     }
00728     return xy / x2;
00729 }
