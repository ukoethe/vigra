00060 /** \brief Apply rank order filter with disc structuring function to the image.
00061

00155 template <class SrcIterator, class SrcAccessor,
00156           class DestIterator, class DestAccessor>
00157 void
00158 discRankOrderFilter(SrcIterator upperleft1, 
00159                     SrcIterator lowerright1, SrcAccessor sa,
00160                     DestIterator upperleft2, DestAccessor da,
00161                     int radius, float rank)
00162 {
00163     vigra_precondition((rank >= 0.0) && (rank <= 1.0),
00164             "discRankOrderFilter(): Rank must be between 0 and 1"
00165             " (inclusive).");
00166     
00167     vigra_precondition(radius >= 0,
00168             "discRankOrderFilter(): Radius must be >= 0.");
00169     
00170     int i, x, y, xmax, ymax, xx, yy;
00171     int rankpos, winsize, leftsum;
00172     
00173     long hist[256];
00174     
00175     // prepare structuring function
00176     std::vector<int> struct_function(radius+1);
00177     struct_function[0] = radius;
00178     
00179     double r2 = (double)radius*radius;
00180     for(i=1; i<=radius; ++i)
00181     {
00182         double r = (double) i - 0.5;
00183         struct_function[i] = (int)(VIGRA_CSTD::sqrt(r2 - r*r) + 0.5);
00184     }
00185 
00186     int w = lowerright1.x - upperleft1.x;
00187     int h = lowerright1.y - upperleft1.y;
00188     
00189     SrcIterator ys(upperleft1);
00190     DestIterator yd(upperleft2);
00191 
00192     for(y=0; y<h; ++y, ++ys.y, ++yd.y)
00193     {
00194         SrcIterator xs(ys);
00195         DestIterator xd(yd);
00196         
00197         // first column
00198         int x0 = 0;
00199         int y0 = y;
00200         int x1 = w - 1;
00201         int y1 = h - y - 1;
00202 
00203         // clear histogram
00204         for(i=0; i<256; ++i) hist[i] = 0;
00205         winsize = 0;
00206         
00207         // init histogram
00208         ymax = (y1 < radius) ? y1 : radius;
00209         for(yy=0; yy<=ymax; ++yy)
00210         {
00211             xmax = (x1 < struct_function[yy]) ? x1 : struct_function[yy];
00212             for(xx=0; xx<=xmax; ++xx)
00213             {
00214                 hist[sa(xs, Diff2D(xx, yy))]++;
00215                 winsize++;
00216             }
00217         }
00218         
00219         ymax = (y0 < radius) ? y0 : radius;
00220         for(yy=1; yy<=ymax; ++yy)
00221         {
00222             xmax = (x1 < struct_function[yy]) ? x1 : struct_function[yy];
00223             for(xx=0; xx<=xmax; ++xx)
00224             {
00225                 hist[sa(xs, Diff2D(xx, -yy))]++;
00226                 winsize++;
00227             }
00228         }
00229     
00230         // find the desired histogram bin 
00231         leftsum = 0;
00232         if(rank == 0.0)
00233         {
00234             for(i=0; i<256; i++)
00235             {
00236                 if(hist[i]) break;
00237             }
00238             rankpos = i;
00239         }
00240         else
00241         {
00242             for(i=0; i<256; i++)
00243             {
00244                 if((float)(hist[i]+leftsum) / winsize >= rank) break;
00245                 leftsum += hist[i];
00246             }
00247             rankpos = i;
00248         }
00249         
00250         da.set(rankpos, xd);
00251         
00252         ++xs.x;
00253         ++xd.x;
00254         
00255         // inner columns
00256         for(x=1; x<w; ++x, ++xs.x, ++xd.x)
00257         {
00258             x0 = x;
00259             y0 = y;
00260             x1 = w - x - 1;
00261             y1 = h - y - 1;
00262             
00263             // update histogram 
00264             // remove pixels at left border 
00265             yy = (y1 < radius) ? y1 : radius;
00266             for(; yy>=0; yy--)
00267             {
00268                 unsigned char cur;
00269                 xx = struct_function[yy]+1;
00270                 if(xx > x0) break;
00271                 
00272                 cur = sa(xs, Diff2D(-xx, yy));
00273                 
00274                 hist[cur]--;
00275                 if(cur < rankpos) leftsum--;
00276                 winsize--;
00277             }
00278             yy = (y0 < radius) ? y0 : radius;
00279             for(; yy>=1; yy--)
00280             {
00281                 unsigned char cur;
00282                 xx = struct_function[yy]+1;
00283                 if(xx > x0) break;
00284                 
00285                 cur = sa(xs, Diff2D(-xx, -yy));
00286                 
00287                 hist[cur]--;
00288                 if(cur < rankpos) leftsum--;
00289                 winsize--;
00290             }
00291             
00292             // add pixels at right border 
00293             yy = (y1 < radius) ? y1 : radius;
00294             for(; yy>=0; yy--)
00295             {
00296                 unsigned char cur;
00297                 xx = struct_function[yy];
00298                 if(xx > x1) break;
00299                 
00300                 cur = sa(xs, Diff2D(xx, yy));
00301                 
00302                 hist[cur]++;
00303                 if(cur < rankpos) leftsum++;
00304                 winsize++;
00305             }
00306             yy = (y0 < radius) ? y0 : radius;
00307             for(; yy>=1; yy--)
00308             {
00309                 unsigned char cur;
00310                 xx = struct_function[yy];
00311                 if(xx > x1) break;
00312                 
00313                 cur = sa(xs, Diff2D(xx, -yy));
00314                 
00315                 hist[cur]++;
00316                 if(cur < rankpos) leftsum++;
00317                 winsize++;
00318             }
00319         
00320             // find the desired histogram bin 
00321             if(rank == 0.0)
00322             {
00323                 if(leftsum == 0)
00324                 {
00325                     // search to the right 
00326                     for(i=rankpos; i<256; i++)
00327                     {
00328                         if(hist[i]) break;
00329                     }
00330                     rankpos = i;
00331                 }
00332                 else
00333                 {
00334                     // search to the left 
00335                     for(i=rankpos-1; i>=0; i--)
00336                     {
00337                         leftsum -= hist[i];
00338                         if(leftsum == 0) break;
00339                     }
00340                     rankpos = i;
00341                 }
00342             }
00343             else  // rank > 0.0 
00344             {
00345                 if((float)leftsum / winsize < rank)
00346                 {
00347                     // search to the right 
00348                     for(i=rankpos; i<256; i++)
00349                     {
00350                         if((float)(hist[i]+leftsum) / winsize >= rank) break;
00351                         leftsum+=hist[i];
00352                     }
00353                     rankpos = i;
00354                 }
00355                 else
00356                 {
00357                     // search to the left 
00358                     for(i=rankpos-1; i>=0; i--)
00359                     {
00360                         leftsum-=hist[i];
00361                         if((float)leftsum / winsize < rank) break;
00362                     }
00363                     rankpos = i;
00364                 }
00365             }
00366                     
00367             da.set(rankpos, xd);
00368         }
00369     }
00370 }

00800 discRankOrderFilterWithMask(SrcIterator upperleft1, 
00801                             SrcIterator lowerright1, SrcAccessor sa,
00802                             MaskIterator upperleftm, MaskAccessor mask,
00803                             DestIterator upperleft2, DestAccessor da,
00804                             int radius, float rank)
00805 {
00806     vigra_precondition((rank >= 0.0) && (rank <= 1.0),
00807                  "discRankOrderFilter(): Rank must be between 0 and 1"
00808                  " (inclusive).");
00809     
00810     vigra_precondition(radius >= 0, "discRankOrderFilter(): Radius must be >= 0.");
00811     
00812     int i, x, y, xmax, ymax, xx, yy;
00813     int rankpos, winsize, leftsum;
00814     
00815     long hist[256];
00816     
00817     // prepare structuring function
00818     std::vector<int> struct_function(radius+1);
00819     struct_function[0] = radius;
00820     
00821     double r2 = (double)radius*radius;
00822     for(i=1; i<=radius; ++i)
00823     {
00824         double r = (double) i - 0.5;
00825         struct_function[i] = (int)(VIGRA_CSTD::sqrt(r2 - r*r) + 0.5);
00826     }
00827 
00828     int w = lowerright1.x - upperleft1.x;
00829     int h = lowerright1.y - upperleft1.y;
00830         
00831     SrcIterator ys(upperleft1);
00832     MaskIterator ym(upperleftm);
00833     DestIterator yd(upperleft2);
00834 
00835     for(y=0; y<h; ++y, ++ys.y, ++yd.y, ++ym.y)
00836     {
00837         SrcIterator xs(ys);
00838         MaskIterator xm(ym);
00839         DestIterator xd(yd);
00840         
00841         // first column
00842         int x0 = 0;
00843         int y0 = y;
00844         int x1 = w - 1;
00845         int y1 = h - y - 1;
00846 
00847         // clear histogram
00848         for(i=0; i<256; ++i) hist[i] = 0;
00849         winsize = 0;
00850         leftsum = 0;
00851         rankpos = 0;
00852         
00853         // init histogram
00854         ymax = (y1 < radius) ? y1 : radius;
00855         for(yy=0; yy<=ymax; ++yy)
00856         {
00857             xmax = (x1 < struct_function[yy]) ? x1 : struct_function[yy];
00858             for(xx=0; xx<=xmax; ++xx)
00859             {
00860                 Diff2D pos(xx, yy);
00861                 if(mask(xm, pos))
00862                 {
00863                     hist[sa(xs, pos)]++;
00864                     winsize++;
00865                 }
00866             }
00867         }
00868         
00869         ymax = (y0 < radius) ? y0 : radius;
00870         for(yy=1; yy<=ymax; ++yy)
00871         {
00872             xmax = (x1 < struct_function[yy]) ? x1 : struct_function[yy];
00873             for(xx=0; xx<=xmax; ++xx)
00874             {
00875                 Diff2D pos(xx, -yy);
00876                 if(mask(xm, pos))
00877                 {
00878                     hist[sa(xs, pos)]++;
00879                     winsize++;
00880                 }
00881             }
00882         }
00883     
00884         // find the desired histogram bin 
00885         if(winsize) 
00886         {
00887             if(rank == 0.0)
00888             {
00889                 for(i=0; i<256; i++)
00890                 {
00891                     if(hist[i]) break;
00892                 }
00893                 rankpos = i;
00894             }
00895             else
00896             {
00897                 for(i=0; i<256; i++)
00898                 {
00899                     if((float)(hist[i]+leftsum) / winsize >= rank) break;
00900                     leftsum += hist[i];
00901                 }
00902                 rankpos = i;
00903             }
00904             
00905             da.set(rankpos, xd);
00906         }
00907             
00908         ++xs.x;
00909         ++xd.x;
00910         ++xm.x;
00911         
00912         // inner columns
00913         for(x=1; x<w; ++x, ++xs.x, ++xd.x, ++xm.x)
00914         {
00915             x0 = x;
00916             y0 = y;
00917             x1 = w - x - 1;
00918             y1 = h - y - 1;
00919             
00920             // update histogram 
00921             // remove pixels at left border 
00922             yy = (y1 < radius) ? y1 : radius;
00923             for(; yy>=0; yy--)
00924             {
00925                 unsigned char cur;
00926                 xx = struct_function[yy]+1;
00927                 if(xx > x0) break;
00928                 
00929                 Diff2D pos(-xx, yy);
00930                 if(mask(xm, pos))
00931                 {
00932                     cur = sa(xs, pos);
00933                     
00934                     hist[cur]--;
00935                     if(cur < rankpos) leftsum--;
00936                     winsize--;
00937                 }
00938             }
00939             yy = (y0 < radius) ? y0 : radius;
00940             for(; yy>=1; yy--)
00941             {
00942                 unsigned char cur;
00943                 xx = struct_function[yy]+1;
00944                 if(xx > x0) break;
00945                 
00946                 Diff2D pos(-xx, -yy);
00947                 if(mask(xm, pos))
00948                 {
00949                     cur = sa(xs, pos);
00950                     
00951                     hist[cur]--;
00952                     if(cur < rankpos) leftsum--;
00953                     winsize--;
00954                 }
00955             }
00956             
00957             // add pixels at right border 
00958             yy = (y1 < radius) ? y1 : radius;
00959             for(; yy>=0; yy--)
00960             {
00961                 unsigned char cur;
00962                 xx = struct_function[yy];
00963                 if(xx > x1) break;
00964                 
00965                 Diff2D pos(xx, yy);
00966                 if(mask(xm, pos))
00967                 {
00968                     cur = sa(xs, pos);
00969                     
00970                     hist[cur]++;
00971                     if(cur < rankpos) leftsum++;
00972                     winsize++;
00973                 }
00974             }
00975             yy = (y0 < radius) ? y0 : radius;
00976             for(; yy>=1; yy--)
00977             {
00978                 unsigned char cur;
00979                 xx = struct_function[yy];
00980                 if(xx > x1) break;
00981                 
00982                 Diff2D pos(xx, -yy);
00983                 if(mask(xm, pos))
00984                 {
00985                     cur = sa(xs, pos);
00986                     
00987                     hist[cur]++;
00988                     if(cur < rankpos) leftsum++;
00989                     winsize++;
00990                 }
00991             }
00992         
00993             // find the desired histogram bin 
00994             if(winsize) 
00995             {
00996                 if(rank == 0.0)
00997                 {
00998                     if(leftsum == 0)
00999                     {
01000                         // search to the right 
01001                         for(i=rankpos; i<256; i++)
01002                         {
01003                             if(hist[i]) break;
01004                         }
01005                         rankpos = i;
01006                     }
01007                     else
01008                     {
01009                         // search to the left 
01010                         for(i=rankpos-1; i>=0; i--)
01011                         {
01012                             leftsum -= hist[i];
01013                             if(leftsum == 0) break;
01014                         }
01015                         rankpos = i;
01016                     }
01017                 }
01018                 else  // rank > 0.0 
01019                 {
01020                     if((float)leftsum / winsize < rank)
01021                     {
01022                         // search to the right 
01023                         for(i=rankpos; i<256; i++)
01024                         {
01025                             if((float)(hist[i]+leftsum) / winsize >= rank) break;
01026                             leftsum+=hist[i];
01027                         }
01028                         rankpos = i;
01029                     }
01030                     else
01031                     {
01032                         // search to the left 
01033                         for(i=rankpos-1; i>=0; i--)
01034                         {
01035                             leftsum-=hist[i];
01036                             if((float)leftsum / winsize < rank) break;
01037                         }
01038                         rankpos = i;
01039                     }
01040                 }
01041                         
01042                 da.set(rankpos, xd);
01043             }
01044             else
01045             {
01046                 leftsum = 0;
01047                 rankpos = 0;
01048             }
01049         }
01050     }
