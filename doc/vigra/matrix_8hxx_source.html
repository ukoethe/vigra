02607 void updateCovarianceMatrix(MultiArrayView<2, T1, C1> const & features,
02608                        U & count, MultiArrayView<2, T2, C2> & mean, MultiArrayView<2, T3, C3> & covariance)
02609 {
02610     MultiArrayIndex n = std::max(rowCount(features), columnCount(features));
02611     vigra_precondition(std::min(rowCount(features), columnCount(features)) == 1,
02612           "updateCovarianceMatrix(): Features must be a row or column vector.");
02613     vigra_precondition(mean.shape() == features.shape(),
02614           "updateCovarianceMatrix(): Shape mismatch between feature vector and mean vector.");

02616           "updateCovarianceMatrix(): Shape mismatch between feature vector and covariance matrix.");

02618     // West's algorithm for incremental covariance matrix computation
02619     Matrix<T2> t = features - mean;
02620     ++count;

02622            f1 = 1.0 - f;

02636     }

02638     {
02639         for(MultiArrayIndex k=0; k<n; ++k)

02641             covariance(k, k) += f1*sq(t(k, 0));
02642             for(MultiArrayIndex l=k+1; l<n; ++l)

02644                 covariance(k, l) += f1*t(k, 0)*t(l, 0);
02645                 covariance(l, k) = covariance(k, l);

02647         }

02742 template <class T, class C1, class C2, class C3, class C4>
02743 void
02744 prepareDataImpl(const MultiArrayView<2, T, C1> & A, 
02745                MultiArrayView<2, T, C2> & res, MultiArrayView<2, T, C3> & offset, MultiArrayView<2, T, C4> & scaling, 
02746                DataPreparationGoals goals)
02747 {
02748     MultiArrayIndex m = rowCount(A);
02749     MultiArrayIndex n = columnCount(A);
02750     vigra_precondition(A.shape() == res.shape() && 
02751                        n == columnCount(offset) && 1 == rowCount(offset) &&
02752                        n == columnCount(scaling) && 1 == rowCount(scaling),
02753                        "prepareDataImpl(): Shape mismatch between input and output.");
02754 
02755     if(!goals)
02756     {
02757         res = A;
02758         offset.init(NumericTraits<T>::zero());
02759         scaling.init(NumericTraits<T>::one());
02760         return;
02761     }
02762     
02763     bool zeroMean = (goals & ZeroMean) != 0;
02764     bool unitVariance = (goals & UnitVariance) != 0;
02765     bool unitNorm = (goals & UnitNorm) != 0;
02766     bool unitSum = (goals & UnitSum) != 0;
02767 
02768     if(unitSum)
02769     {
02770         vigra_precondition(goals == UnitSum,
02771              "prepareData(): Unit sum is not compatible with any other data preparation goal.");
02772              
02773         transformMultiArray(srcMultiArrayRange(A), destMultiArrayRange(scaling), FindSum<T>());
02774         
02775         offset.init(NumericTraits<T>::zero());
02776         
02777         for(MultiArrayIndex k=0; k<n; ++k)
02778         {
02779             if(scaling(0, k) != NumericTraits<T>::zero())
02780             {
02781                 scaling(0, k) = NumericTraits<T>::one() / scaling(0, k);
02782                 columnVector(res, k) = columnVector(A, k) * scaling(0, k);
02783             }
02784             else
02785             {
02786                 scaling(0, k) = NumericTraits<T>::one();
02787             }
02788         }
02789         
02790         return;     
02791     }
02792 
02793     vigra_precondition(!(unitVariance && unitNorm),
02794         "prepareData(): Unit variance and unit norm cannot be achieved at the same time.");
02795 
02796     Matrix<T> mean(1, n), sumOfSquaredDifferences(1, n);
02797     detail::columnStatisticsImpl(A, mean, sumOfSquaredDifferences);
02798     
02799     for(MultiArrayIndex k=0; k<n; ++k)
02800     {
02801         T stdDev = std::sqrt(sumOfSquaredDifferences(0, k) / T(m-1));
02802         if(closeAtTolerance(stdDev / mean(0,k), NumericTraits<T>::zero()))
02803             stdDev = NumericTraits<T>::zero();
02804         if(zeroMean && stdDev > NumericTraits<T>::zero()) 
02805         {
02806             columnVector(res, k) = columnVector(A, k) - mean(0,k);
02807             offset(0, k) = mean(0, k);
02808             mean(0, k) = NumericTraits<T>::zero();
02809         }
02810         else 
02811         {
02812             columnVector(res, k) = columnVector(A, k);
02813             offset(0, k) = NumericTraits<T>::zero();
02814         }
02815         
02816         T norm = mean(0,k) == NumericTraits<T>::zero()
02817                   ? std::sqrt(sumOfSquaredDifferences(0, k))
02818                   : std::sqrt(sumOfSquaredDifferences(0, k) + T(m) * sq(mean(0,k)));
02819         if(unitNorm && norm > NumericTraits<T>::zero())
02820         {
02821             columnVector(res, k) /= norm;
02822             scaling(0, k) = NumericTraits<T>::one() / norm;
02823         }
02824         else if(unitVariance && stdDev > NumericTraits<T>::zero())
02825         {
02826             columnVector(res, k) /= stdDev;
02827             scaling(0, k) = NumericTraits<T>::one() / stdDev;
02828         }
02829         else
02830         {
02831             scaling(0, k) = NumericTraits<T>::one();
02832         }
02833     }
02834 }
