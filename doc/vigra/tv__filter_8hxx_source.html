00074 /** \brief Performs standard Total Variation Regularization
00075 
00076 The algorithm minimizes
00077     
00078 \f[
00079        \min_u \int_\Omega \frac{1}{2} (u-f)^2\;dx + \alpha TV(u)\qquad\qquad (1)
00080 \f]
00081 where <em>\f$ f=f(x)\f$</em> are the two dimensional noisy data,
00082 <em> \f$ u=u(x)\f$</em> are the smoothed data,<em>\f$ \alpha \ge 0 \f$</em>
00083 is the filter parameter and <em>\f$ TV(u)\f$ </em> is the total variation semi-norm.
00084         
00085 <b> Declarations:</b>
00086    
00087 \code
00088 namespace vigra {
00089       template <class stride1,class stride2>
00090       void totalVariationFilter(MultiArrayView<2,double,stride1> data, 
00091                                 MultiArrayView<2,double,stride2> out,
00092                                 double alpha, 
00093                                 int steps, 
00094                                 double eps=0);
00095       void totalVariationFilter(MultiArrayView<2,double,stride1> data,
00096                                 MultiArrayView<2,double,stride2> weight,
00097                                 MultiArrayView<2,double,stride3> out,
00098                                 double alpha, 
00099                                 int steps, 
00100                                 double eps=0);
00101 }
00102 \endcode
00103             
00104 \ref totalVariationFilter() implements a primal-dual algorithm to solve (1).
00105      
00106 Input:
00107      <table>     
00108      <tr><td><em>data</em>:  </td><td> input data to be smoothed. </td></tr>
00109      <tr><td><em>alpha</em>: </td><td> smoothing parameter.</td></tr>
00110      <tr><td><em>steps</em>: </td><td> maximal number of iteration steps. </td></tr>
00111      <tr><td><em>eps</em>:   </td><td> The algorithm stops, if the primal-dual gap is below the threshold <em>eps</em>.

00141 template <class stride1,class stride2>
00142 void totalVariationFilter(MultiArrayView<2,double,stride1> data,MultiArrayView<2,double,stride2> out, double alpha, int steps, double eps=0){
00143 
00144   using namespace multi_math;
00145   int width=data.shape(0),height=data.shape(1);
00146   
00147   MultiArray<2,double> temp1(data.shape()),temp2(data.shape()),vx(data.shape()),vy(data.shape()),u_bar(data.shape());
00148   Kernel1D<double> Lx,LTx;
00149   Lx.initExplicitly(-1,0)=1,-1;                       // = Right sided finite differences for d/dx and d/dy
00150   Lx.setBorderTreatment(BORDER_TREATMENT_REFLECT);   //   with hom. Neumann boundary conditions
00151   LTx.initExplicitly(0,1)=-1,1;                     //  = Left sided finite differences for -d/dx  and -d/dy
00152   LTx.setBorderTreatment(BORDER_TREATMENT_ZEROPAD);  //   with hom. Dirichlet b. c.
00153 
00154   out=data;
00155   u_bar=data;
00156   
00157   double tau=1.0 / std::max(alpha,1.) / std::sqrt(8.0) * 0.06;
00158   double sigma=1.0 / std::sqrt(8.0) / 0.06;
00159     
00160   for (int i=0;i<steps;i++){
00161   
00162     separableConvolveX(srcImageRange(u_bar),destImage(temp1),kernel1d(Lx));setZeroX(temp1);
00163     vx+=(sigma*temp1);
00164     separableConvolveY(srcImageRange(u_bar),destImage(temp1),kernel1d(Lx));setZeroY(temp1);
00165     vy+=(sigma*temp1);
00166     
00167     //project to constraint set
00168     for (int y=0;y<data.shape(1);y++){
00169       for (int x=0;x<data.shape(0);x++){
00170         double l=hypot(vx(x,y),vy(x,y));
00171         if (l>1){
00172           vx(x,y)/=l;
00173           vy(x,y)/=l;
00174         }
00175       }
00176     }
00177     
00178     separableConvolveX(srcImageRange(vx),destImage(temp1),kernel1d(LTx));
00179     separableConvolveY(srcImageRange(vy),destImage(temp2),kernel1d(LTx));
00180     u_bar=out;
00181     out-=tau*(out-data+alpha*(temp1+temp2));
00182     u_bar=2*out-u_bar;   //cf. Chambolle/Pock and Popov's algorithm
00183     
00184     
00185     //stopping criterion
00186     if (eps>0){
00187       separableConvolveX(srcImageRange(out),destImage(temp1),kernel1d(Lx));setZeroX(temp1);
00188       separableConvolveY(srcImageRange(out),destImage(temp2),kernel1d(Lx));setZeroY(temp2);
00189       
00190       double f_primal=0,f_dual=0;
00191       for (int y=0;y<data.shape(1);y++){
00192         for (int x=0;x<data.shape(0);x++){
00193           f_primal+=.5*(out(x,y)-data(x,y))*(out(x,y)-data(x,y))+alpha*hypot(temp1(x,y),temp2(x,y));
00194         }
00195       }
00196       separableConvolveX(srcImageRange(vx),destImage(temp1),kernel1d(LTx));
00197       separableConvolveY(srcImageRange(vy),destImage(temp2),kernel1d(LTx));
00198       for (int y=0;y<data.shape(1);y++){
00199         for (int x=0;x<data.shape(0);x++){
00200           double divv=temp1(x,y)+temp2(x,y);
00201           f_dual+=-.5*alpha*alpha*(divv*divv)+alpha*data(x,y)*divv;
00202         }
00203       }
00204       if (f_primal>0 && (f_primal-f_dual)/f_primal<eps){
00205         break;
00206       }
00207     }
00208   }
00209 }

00289 /** \brief Sets up directional data for anisotropic regularization
00290 
00291 This routine provides a two-dimensional normalized vector field \f$ v \f$, which is normal to edges in the given data,
00292 found as the eigenvector of the structure tensor belonging to the largest eigenvalue.
00293 \f$ v \f$ is encoded by a scalar field \f$ \varphi \f$ of angles, i.e. 
00294 \f$ v(x)=(\cos(\varphi(x)),\sin(\varphi(x)))^\top \f$.
00295 
00296 In addition, two scalar fields \f$ \alpha \f$ and  \f$ \beta \f$ are generated from
00297 scalar parameters \f$ \alpha_{par}\f$ and \f$ \beta_{par}\f$, such that
00298 <center>
00299 <table>
00300 <tr> <td>\f$ \alpha(x)= \alpha_{par}\f$ at edges,</td></tr>
00301 <tr> <td>\f$ \alpha(x)= \beta_{par}\f$ in homogeneous regions,</td></tr>
00302 <tr> <td>\f$ \beta(x)=\beta_{par}\f$ .</td></tr>
00303 </table>
00304 </center>
00305 
00306 <b> Declarations:</b>
00307 
00308 \code
00309 namespace vigra {
00310 void getAnisotropy(MultiArrayView<2,double,stride1> data,
00311                    MultiArrayView<2,double,stride2> phi,
00312                    MultiArrayView<2,double,stride3> alpha, 
00313                    MultiArrayView<2,double,stride4> beta, 
00314                    double alpha_par, 
00315                    double beta_par, 
00316                    double sigma_par, 
00317                    double rho_par, 
00318                    double K_par);
00319 }
00320 \endcode
00321 
00322 Output:
00323 <table>
00324   <tr><td>Three scalar fields <em>phi</em>, <em>alpha</em> and <em>beta</em>.</td></tr>
00325 </table>
00326 
00327 Input:
00328 <table>
00329 <tr><td><em>data</em>:</td><td>two-dimensional scalar field.</td></tr>
00330 <tr><td><em>alpha_par,beta_par</em>:</td><td>two positive values for setting up the scalar fields alpha and beta</tr>
00331 <tr><td><em>sigma_par</em>:</td><td> non-negative parameter for presmoothing the data.</td></tr>
00332 <tr><td> <em>rho_par</em>:</td><td> non-negative parameter for presmoothing the structure tensor.</td></tr>
00333 <tr><td><em>K_par</em>:</td><td> positive edge sensitivity parameter.</td></tr>
00334  </table>
00335  
00336 (see \ref anisotropicTotalVariationFilter() and \ref secondOrderTotalVariationFilter() for usage in an application).
00337 */
00338 doxygen_overloaded_function(template <...> void getAnisotropy)

00340 template <class stride1,class stride2,class stride3,class stride4>
00341 void getAnisotropy(MultiArrayView<2,double,stride1> data,MultiArrayView<2,double,stride2> phi,
00342                     MultiArrayView<2,double,stride3> alpha, MultiArrayView<2,double,stride4> beta, 
00343                     double alpha_par, double beta_par, double sigma_par, double rho_par, double K_par){
00344   
00345   using namespace multi_math;
00346   int width=data.shape(0),height=data.shape(1);
00347   
00348   MultiArray<2,double> smooth(data.shape()),tmp(data.shape());
00349   vigra::Kernel1D<double> gauss;
00350 
00351   
00352   gauss.initGaussian(sigma_par);
00353   separableConvolveX(srcImageRange(data), destImage(tmp), kernel1d(gauss));
00354   separableConvolveY(srcImageRange(tmp), destImage(smooth), kernel1d(gauss));
00355   
00356   MultiArray<2,double> stxx(data.shape()),stxy(data.shape()),styy(data.shape());
00357   
00358   // calculate Structure Tensor at inner scale = sigma and outer scale = rho
00359   vigra::structureTensor(srcImageRange(smooth),destImage(stxx), destImage(stxy), destImage(styy),1.,1.);
00360   
00361   gauss.initGaussian(rho_par);
00362   separableConvolveX(srcImageRange(stxx), destImage(tmp), kernel1d(gauss));
00363   separableConvolveY(srcImageRange(tmp), destImage(stxx), kernel1d(gauss));
00364   separableConvolveX(srcImageRange(stxy), destImage(tmp), kernel1d(gauss));
00365   separableConvolveY(srcImageRange(tmp), destImage(stxy), kernel1d(gauss));
00366   separableConvolveX(srcImageRange(styy), destImage(tmp), kernel1d(gauss));
00367   separableConvolveY(srcImageRange(tmp), destImage(styy), kernel1d(gauss));
00368   
00369   MultiArray<2,double> matrix(Shape2(2,2)),ev(Shape2(2,2)),ew(Shape2(2,1));
00370   
00371    for (int y=0;y<data.shape(1);y++){
00372     for (int x=0;x<data.shape(0);x++){
00373       
00374       matrix(0,0)=stxx(x,y);
00375       matrix(1,1)=styy(x,y);
00376       matrix(0,1)=stxy(x,y);
00377       matrix(1,0)=stxy(x,y);
00378       vigra::linalg::detail::symmetricEigensystem2x2(matrix,ew,ev);
00379       
00380       phi(x,y)=std::atan2(ev(1,0),ev(0,0));
00381       double coherence=ew(0,0)-ew(1,0); 
00382       double c=std::min(K_par*coherence,1.);
00383       alpha(x,y)=alpha_par*c+(1-c)*beta_par;
00384       beta(x,y)=beta_par;
00385       }
00386   }
00387 }

00395 /** \brief Performs Anisotropic Total Variation Regularization
00396 
00397 The algorithm minimizes
00398 \f[
00399 \min_u \int_\Omega \frac{1}{2} (u-f)^2 + \sqrt{\nabla u^\top A \nabla u}\;dx\qquad\qquad(2)
00400 \f]
00401 
00402 where <em> \f$ f=f(x)\f$ </em> are the noisy data, <em> \f$ u=u(x)\f$ </em> are the smoothed data,<em>\f$ \nabla u \f$ </em>
00403 is the image gradient in the sense of Total Variation and <em>\f$ A \f$ </em> is a locally varying symmetric, positive definite  2x2 matrix.
00404 
00405 Matrix <em>\f$ A \f$ </em> is described by providing  for each data point a normalized eigenvector (via angle \f$ \phi \f$)
00406 and two eigenvalues \f$ \alpha>0 \f$ and \f$ \beta>0 \f$.
00407 
00408 \ref getAnisotropy() can be use to set up such data \f$ \phi,\alpha,\beta \f$ by providing a vector field normal to edges.
00409 
00410 <b> Declarations:</b>
00411 
00412 \code

00414   template <class stride1,class stride2,class stride3,class stride4,class stride5,class stride6>
00415   void anisotropicTotalVariationFilter(MultiArrayView<2,double,stride1> data,
00416                                        MultiArrayView<2,double,stride2> weight,
00417                                        MultiArrayView<2,double,stride3> phi,
00418                                        MultiArrayView<2,double,stride4> alpha, 
00419                                        MultiArrayView<2,double,stride5> beta,
00420                                        MultiArrayView<2,double,stride6> out,
00421                                        int steps);
00422 }
00423 \endcode
00424 
00425 \ref anisotropicTotalVariationFilter() implements a primal-dual algorithm to solve (2).
00426 
00427 Input:
00428 <table>
00429 <tr><td><em>data</em>:</td><td>input data to be filtered. </td></tr>
00430 <tr><td><em>steps</em>:</td><td>iteration steps.</td></tr>
00431 <tr><td><em>weight</em> :</td><td>a point-wise weight (\f$ \ge 0 \f$ ) for the data term.</td></tr>
00432 <tr><td><em>phi</em>,<em>alpha</em> and <em>beta</em> :</td><td> describe matrix \f$ A \f$, see above.</td></tr>
00433 </table>
00434 
00435 Output:
00436 <table>
00437 <tr><td><em>out</em> :</td><td>contains filtered data.</td></tr>
00438 </table>
00439 
00440 <b> Usage:</b>
00441 
00442 E.g. with a solution-dependent adaptivity cf. [1], by updating the matrix \f$ A=A(u)\f$
00443 in an outer loop:
00444

00461   anisotropicTotalVariationFilter(data,weight,phi,alpha,beta,out,inner_steps);
00462   }
00463 \endcode

00537 /** \brief Performs Anisotropic Total Variation Regularization
00538 
00539 The algorithm minimizes
00540 
00541 \f[
00542 \min_u \int_\Omega \frac{1}{2} (u-f)^2 + \sqrt{\nabla u^\top A \nabla u}  + \gamma |Hu|_F\;dx \qquad\qquad (3)
00543 \f]
00544 where <em> \f$ f=f(x)\f$ </em> are the noisy data, <em> \f$ u=u(x)\f$ </em> are the smoothed data,<em>\f$ \nabla u \f$ </em>
00545 is the image gradient in the sense of Total Variation, <em>\f$ A \f$ </em> is a locally varying 
00546 symmetric, positive-definite  2x2 matrix and <em>\f$ |Hu|_F \f$</em> is the Frobenius norm of the Hessian of \f$ u \f$.
00547 
00548 Matrix <em>\f$ A \f$ </em> is described by providing  for each data point a normalized eigenvector (via angle \f$ \phi \f$)
00549 and two eigenvalues \f$ \alpha>0 \f$ and \f$ \beta>0 \f$.
00550 \ref getAnisotropy() can be use to set up such data \f$ \phi,\alpha, \beta \f$ by providing a vector field normal to edges.
00551 
00552 \f$ \gamma>0 \f$ is the locally varying regularization parameter for second order.
